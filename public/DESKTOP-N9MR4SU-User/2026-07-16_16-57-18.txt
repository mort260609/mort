  #!/usr/bin/env bash
  
  ROOT="${1:-squashfs-root}"
  
  if [[ ! -d "$ROOT" ]]; then
      echo "Usage: $0 <squashfs-root>"
      exit 1
  fi
  
  green='\033[1;32m'
  yellow='\033[1;33m'
  red='\033[1;31m'
  reset='\033[0m'
  
  ok()   { echo -e "${green}[+]${reset} $*"; }
  warn() { echo -e "${yellow}[!]${reset} $*"; }
  bad()  { echo -e "${red}[-]${reset} $*"; }
  
  echo "==== Firmware inspection ===="
  echo
  
  ########################################
  echo "== build.prop =="
  
  if [[ -f "$ROOT/build.prop" ]]; then
      grep -E \
          'ro.build.tags|ro.board.platform|ro.product.device|ro.product.model|ro.build.version.release|ro.build.display.id' \
          "$ROOT/build.prop"
  
      grep -q 'test-keys' "$ROOT/build.prop" && ok "Built with test-keys"
  else
      bad "build.prop not found"
  fi
  
  echo

  echo "== SSH =="
  
  if [[ -f "$ROOT/usr/local/sbin/sshd" || -f "$ROOT/usr/bin/sshd" ]]; then
      ok "OpenSSH server present"
  else
      bad "No sshd binary"
  fi

      grep -E \
          'PermitRootLogin|PermitEmptyPasswords|PasswordAuthentication|AuthorizedKeysFile|HostKey' \
          "$CFG"
  else
      warn "No sshd_config"
  fi
  
  echo
  
  ########################################
  echo "== passwd =="
  
  if [[ -f "$ROOT/etc/passwd" ]]; then
      grep '^root:' "$ROOT/etc/passwd"
      grep '^sshd:' "$ROOT/etc/passwd"
  
      if grep -q '^root::' "$ROOT/etc/passwd"; then
          ok "Root has empty password field"
      fi
  fi
  
  echo
  
  ########################################
  echo "== rc5 =="
  
  if [[ -f "$ROOT/etc/init.d/rc5" ]]; then
      grep -nE 'ssh|telnet' "$ROOT/etc/init.d/rc5"
  fi
  
  echo
  
  ########################################
  echo "== inetd =="

      cat "$ROOT/etc/inetd.conf"
  fi
