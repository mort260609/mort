  -- Nexora System — Phase 1 Database Schema
  -- Tables: users, user_stats, quests
  -- This mirrors the schema already live in Supabase (planned in an earlier
  -- session), plus two additive Phase 1 patches:
  --   1. users.constraints (equipment/resource constraints, build prompt §4 —
  --      not in the original design, needed for the Data Input screen)
  --   2. an INSERT policy on users (the original design had select/update
  --      only — with no way for a client to ever create its own row, sign-up
  --      could never complete)
  -- Every statement below is safe to run against the already-live database:
  -- CREATE TABLE / ADD CONSTRAINT are guarded with IF NOT EXISTS, and the
  -- policy creation is wrapped so it no-ops if already present.
  
  -- ============================================================
  -- USERS
  -- Extends Supabase's built-in auth.users with Nexora-specific profile data.
  -- ============================================================
  create table if not exists public.users (
    id uuid primary key references auth.users(id) on delete cascade,
    username text unique not null,
    name text not null,
    level int not null default 1,
    exp int not null default 0,
    exp_to_next_level int not null default 100,
    rank text not null default 'E' check (rank in ('E', 'D', 'C', 'B', 'A', 'S')),
    body_weight numeric,
    time_available_minutes int,           -- daily time constraint from onboarding
    main_quest_id uuid,                   -- set after main quest is created (fk added below)
    streak_count int not null default 0,
    streak_freezes_available int not null default 0,
    points_balance int not null default 0, -- NXR balance
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now()
  );
  
  -- Phase 1 patch: optional equipment/resource constraints from Data Input.
  alter table public.users add column if not exists constraints text;
  
  -- ============================================================
  -- USER STATS
  -- Core RPG stats (STR, DEX, INT, VIT, WIS, etc.) — normalized so new
  -- stat types can be added without a schema migration.
  -- ============================================================
  create table if not exists public.user_stats (
    id uuid primary key default gen_random_uuid(),
    user_id uuid not null references public.users(id) on delete cascade,
    stat_key text not null,               -- e.g. 'STR', 'DEX', 'INT', 'VIT'
    value int not null default 0,
    updated_at timestamptz not null default now(),
    unique (user_id, stat_key)
  );
  
  -- ============================================================
  -- QUESTS
  -- Covers Main, Side, and Daily quests via the `tier` column.
  -- Only one active Main/Side/Daily quest per user is expected at a time —
  -- enforce that in application logic (or add a partial unique index later).
  -- ============================================================
  create table if not exists public.quests (
    id uuid primary key default gen_random_uuid(),
    user_id uuid not null references public.users(id) on delete cascade,
    tier text not null check (tier in ('main', 'side', 'daily')),
    title text not null,
    description text not null,
    status text not null default 'active'
      check (status in ('active', 'completed', 'failed', 'grace_period')),
    progress numeric not null default 0,   -- 0-100, used for Side Quest 50% grace rule
    reward_points int not null default 0,
    reward_exp int not null default 0,
    verification_method text not null default 'honor'
      check (verification_method in ('honor', 'ai_review', 'integration')),
    deadline timestamptz,                  -- null for Main Quest (open-ended until done)
    created_at timestamptz not null default now(),
    completed_at timestamptz
  );
  
  do $$
  begin
    if not exists (
      select 1 from pg_constraint where conname = 'fk_main_quest'
    ) then
      alter table public.users
        add constraint fk_main_quest foreign key (main_quest_id) references public.quests(id);
    end if;
  end $$;
  
  -- ============================================================
  -- INDEXES
  -- ============================================================
  create index if not exists idx_quests_user_tier_status on public.quests (user_id, tier, status);
  create index if not exists idx_user_stats_user on public.user_stats (user_id);
  
  -- ============================================================
  -- ROW LEVEL SECURITY
  -- Users can only read/write their own data.
  -- ============================================================
  alter table public.users enable row level security;
  alter table public.user_stats enable row level security;
  alter table public.quests enable row level security;
  
  do $$
  begin
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'users' and policyname = 'Users can view own profile') then
      create policy "Users can view own profile" on public.users for select using (auth.uid() = id);
    end if;
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'users' and policyname = 'Users can update own profile') then
      create policy "Users can update own profile" on public.users for update using (auth.uid() = id);
    end if;
    -- Phase 1 patch: without this, no client can ever create its own profile
    -- row, since the original design had no INSERT policy on users at all.
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'users' and policyname = 'Users can insert own profile') then
      create policy "Users can insert own profile" on public.users for insert with check (auth.uid() = id);
    end if;
  
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'user_stats' and policyname = 'Users can view own stats') then
      create policy "Users can view own stats" on public.user_stats for select using (auth.uid() = user_id);
    end if;
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'user_stats' and policyname = 'Users can update own stats') then
      create policy "Users can update own stats" on public.user_stats for update using (auth.uid() = user_id);
    end if;
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'user_stats' and policyname = 'Users can insert own stats') then
      create policy "Users can insert own stats" on public.user_stats for insert with check (auth.uid() = user_id);
    end if;
  
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'quests' and policyname = 'Users can view own quests') then
      create policy "Users can view own quests" on public.quests for select using (auth.uid() = user_id);
    end if;
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'quests' and policyname = 'Users can update own quests') then
      create policy "Users can update own quests" on public.quests for update using (auth.uid() = user_id);
    end if;
    if not exists (select 1 from pg_policies where schemaname = 'public' and tablename = 'quests' and policyname = 'Users can insert own quests') then
      create policy "Users can insert own quests" on public.quests for insert with check (auth.uid() = user_id);
    end if;
  end $$;
  
  -- Note: quest generation (AI-created quests) should run through a
  -- server-side service role (bypassing RLS), not the client directly,
  -- since the AI Admin — not the player — creates quest rows. This does not
  -- apply to the Phase 1 Main Quest write, which is the player's own
  -- onboarding submission going through their own "insert own quests" policy.

  # Nexora System — Visual Spec Update (v2)
  ### Supersedes PHASE1_BUILD_PROMPT.md §2 and §4. Extracted directly from the working Antigravity demo reference screenshots.
  
  This is more precise than the original spec — use these exact values, not approximations.
  
  ---
  
  ## Global Visual Values
  
  - **Border radius:** 14px on all panels/buttons (not 2-4px as originally spec'd — the working demo reads as sharp-but-not-harsh at 14px, softer than originally described but still clearly geometric, not soft-rounded consumer-app style)
  - **Glow shadow (gold, on hover/active states):** `box-shadow: 0 0 20px rgba(212, 175, 55, 0.4)`
  - **Hover scale:** `transform: scale(1.03)`, 300ms transition
  - **Ambient background particles:** subtle floating particle system, low opacity (2-6%), non-distracting — visible faintly across the whole background, not just inside panels. Purpose is ambient "the System is alive" texture, not a focal effect.
  - **Global tone shift:** background color subtly cycles/breathes over a ~12 second period — very low-amplitude color shift, should read as "alive" not as an obvious animation if you're not looking for it
  - **Typography:** all-caps labels for System/meta text (e.g. "OPERATOR_ALIAS", "SYSTEM_STANDBY"), tracked-out letter-spacing, monospace or near-monospace feel for these labels specifically — this is a bigger visual signature than plain Rajdhani alone. Regular Rajdhani/Inter still applies to primary content text (username value, quest titles, etc.)
  - **Status/diagnostic line pattern:** small dot bullet + all-caps label + value, e.g. `• SYSTEM_STANDBY` / `X_MODE: 0X22DFF` — this terminal-readout style should appear throughout onboarding, not just Screen 1
  
  ---
  
  ## Screen 1 — Initialize System (revised)
  
  Reference: Image 1/2.
  
  - Top-left: `• SYSTEM_STANDBY` (or `• DECRYPTING_CORE` as a loading variant), small, muted
  - Top-right: `X_MODE: 0X22DFF` (or similar system-code readout) — cosmetic, reinforces "real system," value can be arbitrary/randomized
  - Center: `NEXORA` wordmark, large, letter-spaced
  - Subtitle beneath: `LIFE EXECUTION SYSTEM` (this is the actual product tagline — small, muted, tracked-out)
  - Below that: a circular/radial diagnostic element with `STABILITY_CHECK` label and `PASS` status beneath it — this is a nice touch, implies the System just verified itself before letting you in
  - **Primary button: "INITIATE SYSTEM"** — this is the missing piece from the original spec. White/light-filled button (not gold — gold is reserved for active/loading states per Image 2), full-width within the panel, tap-required to proceed (not auto-advancing)
  - On tap: button becomes gold-filled with label changing to `LOADING SEQUENCE...` (Image 2) before transitioning to Screen 2
  - Bottom row: small stat readouts in a row (e.g. `X_5.0.4`, `SEC_77`, `0.4ms`) — cosmetic system-status flavor text, reinforces the "real OS" feel
  - Bottom-right: small circular indicator/dot (status light)
  
  ## Screen 2 — Auth (revised)
  
  Reference: Image 3/4.
  
  - Small label top-left: `• PHASE_01: IDENTITY_INIT`
  - Header: `SYSTEM ACCESS` (large, bold)
  - Thin underline beneath header
  - Field label: `OPERATOR_ALIAS` (small, tracked-out, muted) above the actual input field — this label-above-input pattern should apply to all fields throughout onboarding, not just this one
  - Input field: simple bottom-border or thin full-border style, transparent background
  - **Primary button: "INITIALIZE CORE"** (not "Enter the System" as originally drafted) — disabled/greyed state while field is empty, becomes gold-filled with glow once valid input is present (Image 4)
  - Bottom-left: small animated dots (`• • •`) — likely a subtle "listening/active" indicator
  - Bottom-right: `IDENTIFIER STAGED` — status readout confirming input is ready
  - Very bottom: `• PROTOCOL_SECURITY: ACTIVE` — same terminal-readout pattern as Screen 1
  
  ## Screen 3 — Loading (revised)
  
  Reference: Image 5.
  
  - Much more minimal than originally spec'd — near-fully black screen, no panel at all
  - Centered text: `• INITIALIZING NEXORA OS...` — small, muted, single line (not the three-line typewriter sequence originally spec'd — this version is simpler and arguably more effective, one line is enough)
