  pragma solidity ^0.8.20;
  
  import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
  import {SafeERC20} from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
  import {ReentrancyGuard} from "@openzeppelin/contracts/security/ReentrancyGuard.sol";
  import {NexaiCore} from "./NexaiCore.sol";
  
  /// @dev Minimal slice of Uniswap SwapRouter02's ABI (`IV3SwapRouter`) - only the
  /// exact-output single-hop swap this zapper needs. Note SwapRouter02 omits the
  /// `deadline` field the original SwapRouter has; callers rely on the tx itself
  /// landing or failing rather than a per-call deadline.
  interface ISwapRouter02 {
      struct ExactOutputSingleParams {
          address tokenIn;
          address tokenOut;
          uint24 fee;
          address recipient;
          uint256 amountOut;
          uint256 amountInMaximum;
          uint160 sqrtPriceLimitX96;
      }
  
      function exactOutputSingle(ExactOutputSingleParams calldata params) external payable returns (uint256 amountIn);
  }
  
  /**
   * @title NexaiZapper
   * @dev One-click single-token deposit into NexaiCore, entering via USDC
   * (`zapDepositUsdc`), WETH (`zapDepositWeth`), or USDT (`zapDepositUsdt`). The
   * USDC/WETH paths swap the portion of the deposit needed to cover the *other* basket
   * leg and keep the rest as a direct leg. USDT isn't a basket asset, so its path swaps
   * for both basket legs instead (two swaps, no direct leg). All paths deposit and
   * forward the minted NEXAI to the caller in one transaction. Scoped to the current
   * two-token (WETH/USDC) basket; does not generalize to arbitrary baskets. Holds no
   * funds or state between transactions.
   */
  contract NexaiZapper is ReentrancyGuard {
      using SafeERC20 for IERC20;
  
      ISwapRouter02 public immutable swapRouter;
      NexaiCore public immutable core;
      IERC20 public immutable weth;
      IERC20 public immutable usdc;
      uint24 public immutable swapFee;
      IERC20 public immutable usdt;
      uint24 public immutable usdtWethFee;
      uint24 public immutable usdtUsdcFee;
  
      event ZappedDeposit(address indexed user, uint256 nexaiAmount, uint256 usdcSpent);
      event ZappedDepositWeth(address indexed user, uint256 nexaiAmount, uint256 wethSpent);
      event ZappedDepositUsdt(address indexed user, uint256 nexaiAmount, uint256 usdtSpent);
  
      constructor(
          address _swapRouter,
          address _core,
          address _weth,
          address _usdc,
          uint24 _swapFee,
          address _usdt,
          uint24 _usdtWethFee,
          uint24 _usdtUsdcFee
      ) {
          require(_swapRouter != address(0), "Invalid swap router address");
          require(_core != address(0), "Invalid core address");
          require(_weth != address(0), "Invalid WETH address");
          require(_usdc != address(0), "Invalid USDC address");
          require(_usdt != address(0), "Invalid USDT address");
  
          swapRouter = ISwapRouter02(_swapRouter);
          core = NexaiCore(_core);
          weth = IERC20(_weth);
          usdc = IERC20(_usdc);
          swapFee = _swapFee;
          usdt = IERC20(_usdt);
          usdtWethFee = _usdtWethFee;
          usdtUsdcFee = _usdtUsdcFee;
      }
  
      /**
       * @dev Deposits USDC only and mints NEXAI to the caller.
       * @param nexaiAmount Amount of NEXAI to mint (18 decimals).
       * @param maxUsdcIn Upper bound on total USDC spent (direct leg + swap leg combined).
       * Acts as the slippage guard - reverts if the swap would need more than this covers.
       * @return usdcSpent The actual total USDC spent across both legs.
       */
      function zapDepositUsdc(uint256 nexaiAmount, uint256 maxUsdcIn) external nonReentrant returns (uint256 usdcSpent) {
          require(nexaiAmount > 0, "Amount must be greater than zero");
  
          uint256 wethWeight = core.basketWeights(address(weth));
          uint256 usdcWeight = core.basketWeights(address(usdc));
          require(
              wethWeight > 0 && usdcWeight > 0 && _basketIsExactlyTwoTokens(), "Basket is not WETH/USDC"
          );
  
          // Mirrors NexaiCore.deposit()'s own round-up formula so the amounts approved
          // below match exactly what deposit() will pull - see NexaiCore.sol.
          uint256 requiredWeth = (nexaiAmount * wethWeight + 1e18 - 1) / 1e18;
          uint256 requiredUsdc = (nexaiAmount * usdcWeight + 1e18 - 1) / 1e18;
          require(maxUsdcIn > requiredUsdc, "maxUsdcIn too low to cover direct USDC leg");
  
          usdc.safeTransferFrom(msg.sender, address(this), maxUsdcIn);
  
          uint256 usdcBudgetForSwap = maxUsdcIn - requiredUsdc;
          usdc.forceApprove(address(swapRouter), usdcBudgetForSwap);
          uint256 usdcSpentOnSwap = swapRouter.exactOutputSingle(
              ISwapRouter02.ExactOutputSingleParams({
                  tokenIn: address(usdc),
                  tokenOut: address(weth),
                  fee: swapFee,
                  recipient: address(this),
                  amountOut: requiredWeth,
                  amountInMaximum: usdcBudgetForSwap,
                  sqrtPriceLimitX96: 0
              })
          );
          usdc.forceApprove(address(swapRouter), 0);
  
          weth.forceApprove(address(core), requiredWeth);
          usdc.forceApprove(address(core), requiredUsdc);
          core.deposit(nexaiAmount);
          weth.forceApprove(address(core), 0);
          usdc.forceApprove(address(core), 0);
  
          // core.deposit() minted NEXAI to address(this) (the Zapper is msg.sender from
          // Core's point of view) - forward it on to the actual depositor.
          IERC20(address(core.indexToken())).safeTransfer(msg.sender, nexaiAmount);
  
          usdcSpent = requiredUsdc + usdcSpentOnSwap;
          uint256 refund = maxUsdcIn - usdcSpent;
          if (refund > 0) usdc.safeTransfer(msg.sender, refund);
  
          emit ZappedDeposit(msg.sender, nexaiAmount, usdcSpent);
      }
  
      /**
       * @dev Deposits WETH only and mints NEXAI to the caller. Mirrors zapDepositUsdc,
       * with the direct/swap legs reversed - direct leg is WETH, swap leg buys the USDC
       * needed with WETH.
       * @param nexaiAmount Amount of NEXAI to mint (18 decimals).
       * @param maxWethIn Upper bound on total WETH spent (direct leg + swap leg combined).
       * Acts as the slippage guard - reverts if the swap would need more than this covers.
       * @return wethSpent The actual total WETH spent across both legs.
       */
      function zapDepositWeth(uint256 nexaiAmount, uint256 maxWethIn) external nonReentrant returns (uint256 wethSpent) {
          require(nexaiAmount > 0, "Amount must be greater than zero");
  
          uint256 wethWeight = core.basketWeights(address(weth));
          uint256 usdcWeight = core.basketWeights(address(usdc));
          require(
              wethWeight > 0 && usdcWeight > 0 && _basketIsExactlyTwoTokens(), "Basket is not WETH/USDC"
          );
  
          // Mirrors NexaiCore.deposit()'s own round-up formula so the amounts approved
          // below match exactly what deposit() will pull - see NexaiCore.sol.
          uint256 requiredWeth = (nexaiAmount * wethWeight + 1e18 - 1) / 1e18;
          uint256 requiredUsdc = (nexaiAmount * usdcWeight + 1e18 - 1) / 1e18;
          require(maxWethIn > requiredWeth, "maxWethIn too low to cover direct WETH leg");
  
          weth.safeTransferFrom(msg.sender, address(this), maxWethIn);
  
          uint256 wethBudgetForSwap = maxWethIn - requiredWeth;
          weth.forceApprove(address(swapRouter), wethBudgetForSwap);
          uint256 wethSpentOnSwap = swapRouter.exactOutputSingle(
              ISwapRouter02.ExactOutputSingleParams({
                  tokenIn: address(weth),
                  tokenOut: address(usdc),
                  fee: swapFee,
                  recipient: address(this),
                  amountOut: requiredUsdc,
                  amountInMaximum: wethBudgetForSwap,
                  sqrtPriceLimitX96: 0
              })
          );
          weth.forceApprove(address(swapRouter), 0);
  
          weth.forceApprove(address(core), requiredWeth);
          usdc.forceApprove(address(core), requiredUsdc);
          core.deposit(nexaiAmount);
          weth.forceApprove(address(core), 0);
          usdc.forceApprove(address(core), 0);
  
          // core.deposit() minted NEXAI to address(this) (the Zapper is msg.sender from
          // Core's point of view) - forward it on to the actual depositor.
          IERC20(address(core.indexToken())).safeTransfer(msg.sender, nexaiAmount);
  
          wethSpent = requiredWeth + wethSpentOnSwap;
          uint256 refund = maxWethIn - wethSpent;
          if (refund > 0) weth.safeTransfer(msg.sender, refund);
  
          emit ZappedDepositWeth(msg.sender, nexaiAmount, wethSpent);
      }
  
      /**
       * @dev Deposits USDT only and mints NEXAI to the caller. USDT isn't a basket asset,
       * so unlike zapDepositUsdc/zapDepositWeth there is no direct leg - both requiredWeth
       * and requiredUsdc are bought via separate swaps, both drawn from the same
       * maxUsdtIn budget. The WETH leg swaps first with the *full* budget as its ceiling;
       * whatever remains becomes the USDC leg's ceiling. If the two legs' combined actual
       * cost exceeds maxUsdtIn, the second swap's own slippage check reverts - the same
       * failure mode as the single-swap paths, just triggered by the second leg instead
       * of the first.
       * @param nexaiAmount Amount of NEXAI to mint (18 decimals).
       * @param maxUsdtIn Upper bound on total USDT spent across both swap legs combined.
       * @return usdtSpent The actual total USDT spent across both legs.
       */
      function zapDepositUsdt(uint256 nexaiAmount, uint256 maxUsdtIn) external nonReentrant returns (uint256 usdtSpent) {
          require(nexaiAmount > 0, "Amount must be greater than zero");
  
          uint256 wethWeight = core.basketWeights(address(weth));
          uint256 usdcWeight = core.basketWeights(address(usdc));
          require(
              wethWeight > 0 && usdcWeight > 0 && _basketIsExactlyTwoTokens(), "Basket is not WETH/USDC"
          );
  
          // Mirrors NexaiCore.deposit()'s own round-up formula so the amounts approved
          // below match exactly what deposit() will pull - see NexaiCore.sol.
          uint256 requiredWeth = (nexaiAmount * wethWeight + 1e18 - 1) / 1e18;
          uint256 requiredUsdc = (nexaiAmount * usdcWeight + 1e18 - 1) / 1e18;
  
          usdt.safeTransferFrom(msg.sender, address(this), maxUsdtIn);
  
          // One approval covers both legs - the router's own transferFrom in leg one
          // already decrements the allowance, so leg two automatically sees whatever
          // budget leg one didn't spend without needing a second approve call.
          usdt.forceApprove(address(swapRouter), maxUsdtIn);
  
          uint256 usdtSpentOnWeth = swapRouter.exactOutputSingle(
              ISwapRouter02.ExactOutputSingleParams({
                  tokenIn: address(usdt),
                  tokenOut: address(weth),
                  fee: usdtWethFee,
                  recipient: address(this),
                  amountOut: requiredWeth,
                  amountInMaximum: maxUsdtIn,
                  sqrtPriceLimitX96: 0
              })
          );
  
          uint256 remainingUsdtBudget = maxUsdtIn - usdtSpentOnWeth;
          uint256 usdtSpentOnUsdc = swapRouter.exactOutputSingle(
              ISwapRouter02.ExactOutputSingleParams({
                  tokenIn: address(usdt),
                  tokenOut: address(usdc),
                  fee: usdtUsdcFee,
                  recipient: address(this),
                  amountOut: requiredUsdc,
                  amountInMaximum: remainingUsdtBudget,
                  sqrtPriceLimitX96: 0
              })
          );
          usdt.forceApprove(address(swapRouter), 0);
  
          weth.forceApprove(address(core), requiredWeth);
          usdc.forceApprove(address(core), requiredUsdc);
          core.deposit(nexaiAmount);
          weth.forceApprove(address(core), 0);
          usdc.forceApprove(address(core), 0);
  
          // core.deposit() minted NEXAI to address(this) (the Zapper is msg.sender from
          // Core's point of view) - forward it on to the actual depositor.
          IERC20(address(core.indexToken())).safeTransfer(msg.sender, nexaiAmount);
  
          usdtSpent = usdtSpentOnWeth + usdtSpentOnUsdc;
          uint256 refund = maxUsdtIn - usdtSpent;
          if (refund > 0) usdt.safeTransfer(msg.sender, refund);
  
          emit ZappedDepositUsdt(msg.sender, nexaiAmount, usdtSpent);
      }
  
      /// @dev NexaiCore's `basketTokens` is a public array with no length getter (only
      /// indexed access is auto-generated), so we detect a third basket token by probing
      /// index 2: it reverts with an out-of-bounds panic if the basket has at most two
      /// tokens, and succeeds if a third one exists. Without this, a basket expanded to
      /// WETH + USDC + anything else would still pass the weight checks above, then fail
      /// deep inside core.deposit() with a confusing raw revert instead of a clear one.
      function _basketIsExactlyTwoTokens() private view returns (bool) {
          try core.basketTokens(2) returns (address) {
              return false;
          } catch {
              return true;
          }
      }
  }

  /// @dev Minimal slice of Uniswap V3's NonfungiblePositionManager ABI - same subset

  /**
   * @title SeedUsdtPools
   * @dev Creates and seeds two Uniswap V3 pools for the mock USDT entry token on Base
   * Sepolia: USDT/WETH, priced off the *live* WETH/USDC pool (not the original basket
   * ratio, which has drifted since real swaps executed against that pool - see the
   * frontend's QuoterV2 integration for the same concern), and USDT/USDC, seeded near
   * 1:1 since both are ~$1 stablecoins. Testnet-only.
   */
  contract SeedUsdtPoolsScript is Script {
      address constant FACTORY = 0x4752ba5DBc23f44D87826276BF6Fd6b1C372aD24;
      address constant POSITION_MANAGER = 0x27F971cb582BF9E50F397e4d29a5C7A34f11faA2;
  
      address constant WETH = 0x5dAa95Cc547e98B3CEf227eD1D612CDd3d9E30a6;
      address constant USDC = 0x1a0e689b2b55c4E92571b37e0DA2C3b0EeE30B4a;
      // Set this to the address DeployUsdt.s.sol prints before running this script.
      address constant USDT = address(0);
  
      uint24 constant WETH_USDC_FEE = 3000; // the fee tier SeedPool.s.sol already seeded
      uint24 constant USDT_WETH_FEE = 3000; // 0.3%, same tier as the existing volatile pair
      uint24 constant USDT_USDC_FEE = 500; // 0.05%, standard tier for a stable/stable pair
  
      // Full-range ticks, tick spacing 60 (0.3% tier): floor(887272 / 60) * 60
      int24 constant TICK_LOWER_3000 = -887220;
      int24 constant TICK_UPPER_3000 = 887220;
      // Full-range ticks, tick spacing 10 (0.05% tier): floor(887272 / 10) * 10
      int24 constant TICK_LOWER_500 = -887270;
      int24 constant TICK_UPPER_500 = 887270;
  
      uint256 constant WETH_SEED_FOR_USDT_POOL = 10 * 10 ** 18;
      uint256 constant USDT_USDC_SEED = 50_000 * 10 ** 6; // equal amounts on both sides - 1:1
  
      function run() external {
          require(USDT != address(0), "Set the USDT constant to the deployed mock USDT address first");
  
          uint256 deployerPrivateKey = vm.envUint("PRIVATE_KEY");
          address deployer = vm.addr(deployerPrivateKey);
          vm.startBroadcast(deployerPrivateKey);
  
          uint256 usdcPerWeth = _readUsdcPerWethFromLivePool();
          uint256 usdtSeedForWethPool = (WETH_SEED_FOR_USDT_POOL * usdcPerWeth) / 10 ** 18;
  
          MockERC20(USDT).mint(deployer, usdtSeedForWethPool + USDT_USDC_SEED);
          MockERC20(USDC).mint(deployer, USDT_USDC_SEED);
          MockERC20(WETH).mint(deployer, WETH_SEED_FOR_USDT_POOL);
  
          console.log("Live WETH/USDC price implies USDC per WETH:", usdcPerWeth);
          console.log("Seeding USDT/WETH pool with USDT:", usdtSeedForWethPool);
  
          _seedPool(
              USDT, WETH, USDT_WETH_FEE, usdtSeedForWethPool, WETH_SEED_FOR_USDT_POOL, TICK_LOWER_3000, TICK_UPPER_3000, deployer
          );
          _seedPool(USDT, USDC, USDT_USDC_FEE, USDT_USDC_SEED, USDT_USDC_SEED, TICK_LOWER_500, TICK_UPPER_500, deployer);
  
          vm.stopBroadcast();
      }
  
      /// @dev Reads the existing WETH/USDC pool's current sqrtPriceX96 and returns how much
      /// USDC (6dp) is implied per 1 whole WETH right now, so the USDT/WETH pool tracks the
      /// live price rather than the pool's original (now-stale) seed ratio.
      function _readUsdcPerWethFromLivePool() internal view returns (uint256) {
          address pool = IUniswapV3Factory(FACTORY).getPool(WETH, USDC, WETH_USDC_FEE);
          require(pool != address(0), "WETH/USDC pool not found - run SeedPool.s.sol first");
  
          (uint160 sqrtPriceX96,,,,,,) = IUniswapV3PoolState(pool).slot0();
          uint256 priceX192 = uint256(sqrtPriceX96) * uint256(sqrtPriceX96); // (token1/token0) * 2^192
  
          (address token0,) = WETH < USDC ? (WETH, USDC) : (USDC, WETH);
  
          if (token0 == USDC) {
              // priceX192 = WETH_wei per USDC_wei * 2^192 -> invert for USDC_wei per 1e18 WETH_wei
              return Math.mulDiv(10 ** 18, uint256(1) << 192, priceX192);
          } else {
              // priceX192 = USDC_wei per WETH_wei * 2^192 already
              return Math.mulDiv(priceX192, 10 ** 18, uint256(1) << 192);
          }
      }
  
      function _seedPool(
          address tokenA,
          address tokenB,
          uint24 fee,
          uint256 amountA,
          uint256 amountB,
          int24 tickLower,
          int24 tickUpper,
          address recipient
      ) internal {
          // Same sort-then-assign-reserves pattern as SeedPool.s.sol, generalized to any pair.
          (address token0, address token1) = tokenA < tokenB ? (tokenA, tokenB) : (tokenB, tokenA);
          (uint256 reserve0, uint256 reserve1) = token0 == tokenA ? (amountA, amountB) : (amountB, amountA);
  
          uint160 sqrtPriceX96 = uint160(Math.sqrt(Math.mulDiv(reserve1, uint256(1) << 192, reserve0)));
  
          INonfungiblePositionManager positionManager = INonfungiblePositionManager(POSITION_MANAGER);
  
          address pool = positionManager.createAndInitializePoolIfNecessary(token0, token1, fee, sqrtPriceX96);
          console.log("Pool created/initialized at:", pool);
  
          IERC20(token0).approve(POSITION_MANAGER, reserve0);
          IERC20(token1).approve(POSITION_MANAGER, reserve1);
  
          (uint256 tokenId, uint128 liquidity, uint256 amount0, uint256 amount1) = positionManager.mint(
              INonfungiblePositionManager.MintParams({
                  token0: token0,
                  token1: token1,
                  fee: fee,
                  tickLower: tickLower,
                  tickUpper: tickUpper,
                  amount0Desired: reserve0,
                  amount1Desired: reserve1,
                  amount0Min: 0,
                  amount1Min: 0,
                  recipient: recipient,
                  deadline: block.timestamp + 15 minutes
              })
          );
  
          console.log("Position NFT minted, tokenId:", tokenId);
          console.log("Liquidity:", uint256(liquidity));
          console.log("Amount0 used:", amount0);
          console.log("Amount1 used:", amount1);
      }
  }

  # Frontend: Add USDT as a Third Entry Token Option
  
  The zapper now supports USDC, WETH, and USDT entry, all deployed and
  verified on-chain. New zapper address:
  `0xbfFD94A8846fC8b9Bc5eeB702B31494B5d5B4728`
  
  New USDT-related addresses:
  - Mock USDT: `0x6972023ed96b910BBD6172c21dd0F86375419202`
  - USDT/WETH pool fee tier: 3000 (0.3%)
  - USDT/USDC pool fee tier: 500 (0.05%)
  
  Extend the existing token-toggle pattern (USDC/WETH) to include USDT as
  a third option.
  
  ## Requirements
  
  - Update `VITE_NEXAI_ZAPPER` to the new address; add `VITE_USDT` to `.env`
  - Add USDT to `CONTRACT_ADDRESSES` in `constants.ts`, add `zapDepositUsdt`
    to `ZAPPER_ABI`
  - Extend the toggle to three options: "Pay with USDC" / "Pay with WETH"
    / "Pay with USDT" — same `variant` pattern as the existing two
  - **Cost estimate is different for USDT — it needs TWO quotes, not one**,
    since USDT isn't a basket asset (no direct leg, both legs are swaps):
    - Quote 1: `tokenIn: USDT, tokenOut: WETH, amount: requiredWeth`
    - Quote 2: `tokenIn: USDT, tokenOut: USDC, amount: requiredUsdc`
    - Total estimated USDT cost = sum of both quote results (no separate
      "direct leg" to add, unlike the USDC/WETH paths)
    - Both quotes need their own `swapFee` — read `usdtWethFee` and
      `usdtUsdcFee` from the zapper contract (same on-chain-read pattern
      as the existing `swapFee` read, add two new ABI entries), don't
      hardcode 3000/500 in the frontend
    - Gate both new quote hooks on `zapEntryToken === 'usdt'`, consistent
      with how the existing two quotes are gated — three total quote
      hooks now, but at most two active at once (USDT needs both of its
      own, USDC/WETH each need only one)
  - Approve + deposit mirrors the existing pattern: approve USDT for
    `maxUsdtIn` (same 5% buffer logic applied to the combined two-quote
    total), call `zapDepositUsdt(nexaiAmount, maxUsdtIn)`
  - Add USDT balance and USDT-zapper-allowance to the existing batched
    `useReadContracts` call, same pattern as WETH/USDC
  
  ## Out of Scope
  - No changes to `NexaiZapper.sol` — already deployed, this is
    frontend-only
  - No further entry tokens beyond USDC/WETH/USDT for this trial
