  // SPDX-License-Identifier: MIT
  pragma solidity ^0.8.20;
  
  import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
  import {SafeERC20} from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
  import {ReentrancyGuard} from "@openzeppelin/contracts/security/ReentrancyGuard.sol";
  import {NexaiCore} from "./NexaiCore.sol";
  
  /// @dev Minimal slice of Uniswap SwapRouter02's ABI (`IV3SwapRouter`) - only the
  /// exact-output single-hop swap this zapper needs. Note SwapRouter02 omits the
  /// `deadline` field the original SwapRouter has; callers rely on the tx itself
  /// landing or failing rather than a per-call deadline.
  interface ISwapRouter02 {
      struct ExactOutputSingleParams {
          address tokenIn;
          address tokenOut;
          uint24 fee;
          address recipient;
          uint256 amountOut;
          uint256 amountInMaximum;
          uint160 sqrtPriceLimitX96;
      }
  
      function exactOutputSingle(ExactOutputSingleParams calldata params) external payable returns (uint256 amountIn);
  }
  
  /**
   * @title NexaiZapper
   * @dev One-click single-token (USDC) deposit into NexaiCore. Swaps the portion of the
   * deposit needed to cover the basket's WETH leg via Uniswap V3, keeps the rest as the
   * direct USDC leg, then deposits and forwards the minted NEXAI to the caller - all in
   * one transaction. Scoped to the current two-token (WETH/USDC) basket; does not
   * generalize to arbitrary baskets. Holds no funds or state between transactions.
   */
  contract NexaiZapper is ReentrancyGuard {
      using SafeERC20 for IERC20;
  
      ISwapRouter02 public immutable swapRouter;
      NexaiCore public immutable core;
      IERC20 public immutable weth;
      IERC20 public immutable usdc;
      uint24 public immutable swapFee;
  
      event ZappedDeposit(address indexed user, uint256 nexaiAmount, uint256 usdcSpent);
  
      constructor(address _swapRouter, address _core, address _weth, address _usdc, uint24 _swapFee) {
          require(_swapRouter != address(0), "Invalid swap router address");
          require(_core != address(0), "Invalid core address");
          require(_weth != address(0), "Invalid WETH address");
          require(_usdc != address(0), "Invalid USDC address");
  
          swapRouter = ISwapRouter02(_swapRouter);
          core = NexaiCore(_core);
          weth = IERC20(_weth);
          usdc = IERC20(_usdc);
          swapFee = _swapFee;
      }
  
      /**
       * @dev Deposits USDC only and mints NEXAI to the caller.
       * @param nexaiAmount Amount of NEXAI to mint (18 decimals).
       * @param maxUsdcIn Upper bound on total USDC spent (direct leg + swap leg combined).
       * Acts as the slippage guard - reverts if the swap would need more than this covers.
       * @return usdcSpent The actual total USDC spent across both legs.
       */
      function zapDepositUsdc(uint256 nexaiAmount, uint256 maxUsdcIn) external nonReentrant returns (uint256 usdcSpent) {
          require(nexaiAmount > 0, "Amount must be greater than zero");
  
          uint256 wethWeight = core.basketWeights(address(weth));
          uint256 usdcWeight = core.basketWeights(address(usdc));
          require(wethWeight > 0 && usdcWeight > 0, "Basket is not WETH/USDC");
  
          // Mirrors NexaiCore.deposit()'s own round-up formula so the amounts approved
          // below match exactly what deposit() will pull - see NexaiCore.sol.
          uint256 requiredWeth = (nexaiAmount * wethWeight + 1e18 - 1) / 1e18;
          uint256 requiredUsdc = (nexaiAmount * usdcWeight + 1e18 - 1) / 1e18;
          require(maxUsdcIn > requiredUsdc, "maxUsdcIn too low to cover direct USDC leg");
  
          usdc.safeTransferFrom(msg.sender, address(this), maxUsdcIn);
  
          uint256 usdcBudgetForSwap = maxUsdcIn - requiredUsdc;
          usdc.forceApprove(address(swapRouter), usdcBudgetForSwap);
          uint256 usdcSpentOnSwap = swapRouter.exactOutputSingle(
              ISwapRouter02.ExactOutputSingleParams({
                  tokenIn: address(usdc),
                  tokenOut: address(weth),
                  fee: swapFee,
                  recipient: address(this),
                  amountOut: requiredWeth,
                  amountInMaximum: usdcBudgetForSwap,
                  sqrtPriceLimitX96: 0
              })
          );
          usdc.forceApprove(address(swapRouter), 0);
  
          weth.forceApprove(address(core), requiredWeth);
          usdc.forceApprove(address(core), requiredUsdc);
          core.deposit(nexaiAmount);
          weth.forceApprove(address(core), 0);
          usdc.forceApprove(address(core), 0);
  
          // core.deposit() minted NEXAI to address(this) (the Zapper is msg.sender from
          // Core's point of view) - forward it on to the actual depositor.
          IERC20(address(core.indexToken())).safeTransfer(msg.sender, nexaiAmount);
  
          usdcSpent = requiredUsdc + usdcSpentOnSwap;
          uint256 refund = maxUsdcIn - usdcSpent;
          if (refund > 0) usdc.safeTransfer(msg.sender, refund);
  
          emit ZappedDeposit(msg.sender, nexaiAmount, usdcSpent);
      }
  }

  /// @dev Stands in for Uniswap's SwapRouter02 in unit tests: swaps at a fixed,
  /// configurable USDC-per-WETH rate instead of doing real AMM math, so these tests
  /// exercise the Zapper's own accounting (not Uniswap's), deterministically and
  /// without needing a fork.
  contract MockSwapRouter is ISwapRouter02 {
      using SafeERC20 for IERC20;
  
      uint256 public usdcPerWeth; // USDC (6dp) required per 1e18 WETH out
  
      constructor(uint256 _usdcPerWeth) {
          usdcPerWeth = _usdcPerWeth;
      }
  
      function exactOutputSingle(ExactOutputSingleParams calldata params)
          external
          payable
          override
          returns (uint256 amountIn)
      {
          amountIn = (params.amountOut * usdcPerWeth) / 1e18;
          require(amountIn <= params.amountInMaximum, "Too much requested");
  
          IERC20(params.tokenIn).safeTransferFrom(msg.sender, address(this), amountIn);
          IERC20(params.tokenOut).safeTransfer(params.recipient, params.amountOut);
      }
  }

  /// @dev Stands in for Uniswap's SwapRouter02 in unit tests: swaps at a fixed,
  /// configurable USDC-per-WETH rate instead of doing real AMM math, so these tests
  /// exercise the Zapper's own accounting (not Uniswap's), deterministically and
  /// without needing a fork.
  contract MockSwapRouter is ISwapRouter02 {
      using SafeERC20 for IERC20;
  
      uint256 public usdcPerWeth; // USDC (6dp) required per 1e18 WETH out
  
      constructor(uint256 _usdcPerWeth) {
          usdcPerWeth = _usdcPerWeth;
      }
  
      function exactOutputSingle(ExactOutputSingleParams calldata params)
          external
          payable
          override
          returns (uint256 amountIn)
      {
          amountIn = (params.amountOut * usdcPerWeth) / 1e18;
          require(amountIn <= params.amountInMaximum, "Too much requested");
  
          IERC20(params.tokenIn).safeTransferFrom(msg.sender, address(this), amountIn);
          IERC20(params.tokenOut).safeTransfer(params.recipient, params.amountOut);
      }
  }

  /// @dev Minimal slice of Uniswap SwapRouter02's ABI (`IV3SwapRouter`) - only the
  /// exact-output single-hop swap this zapper needs. Note SwapRouter02 omits the
  /// `deadline` field the original SwapRouter has; callers rely on the tx itself
  /// landing or failing rather than a per-call deadline.
  interface ISwapRouter02 {
      struct ExactOutputSingleParams {
          address tokenIn;
          address tokenOut;
          uint24 fee;
          address recipient;
          uint256 amountOut;
          uint256 amountInMaximum;
          uint160 sqrtPriceLimitX96;
      }
  
      function exactOutputSingle(ExactOutputSingleParams calldata params) external payable returns (uint256 amountIn);
  }
  
  /**
   * @title NexaiZapper
   * @dev One-click single-token (USDC) deposit into NexaiCore. Swaps the portion of the
   * deposit needed to cover the basket's WETH leg via Uniswap V3, keeps the rest as the
   * direct USDC leg, then deposits and forwards the minted NEXAI to the caller - all in
   * one transaction. Scoped to the current two-token (WETH/USDC) basket; does not
   * generalize to arbitrary baskets. Holds no funds or state between transactions.
   */
  contract NexaiZapper is ReentrancyGuard {
      using SafeERC20 for IERC20;
  
      ISwapRouter02 public immutable swapRouter;
      NexaiCore public immutable core;
      IERC20 public immutable weth;
      IERC20 public immutable usdc;
      uint24 public immutable swapFee;
  
      event ZappedDeposit(address indexed user, uint256 nexaiAmount, uint256 usdcSpent);
  
      constructor(address _swapRouter, address _core, address _weth, address _usdc, uint24 _swapFee) {
          require(_swapRouter != address(0), "Invalid swap router address");
          require(_core != address(0), "Invalid core address");
          require(_weth != address(0), "Invalid WETH address");
          require(_usdc != address(0), "Invalid USDC address");
  
          swapRouter = ISwapRouter02(_swapRouter);
          core = NexaiCore(_core);
          weth = IERC20(_weth);
          usdc = IERC20(_usdc);
          swapFee = _swapFee;
      }
  
      /**
       * @dev Deposits USDC only and mints NEXAI to the caller.
       * @param nexaiAmount Amount of NEXAI to mint (18 decimals).
       * @param maxUsdcIn Upper bound on total USDC spent (direct leg + swap leg combined).
       * Acts as the slippage guard - reverts if the swap would need more than this covers.
       * @return usdcSpent The actual total USDC spent across both legs.
       */
      function zapDepositUsdc(uint256 nexaiAmount, uint256 maxUsdcIn) external nonReentrant returns (uint256 usdcSpent) {
          require(nexaiAmount > 0, "Amount must be greater than zero");
  
          uint256 wethWeight = core.basketWeights(address(weth));
          uint256 usdcWeight = core.basketWeights(address(usdc));
          require(
              wethWeight > 0 && usdcWeight > 0 && _basketIsExactlyTwoTokens(), "Basket is not WETH/USDC"
          );
  
          // Mirrors NexaiCore.deposit()'s own round-up formula so the amounts approved
          // below match exactly what deposit() will pull - see NexaiCore.sol.
          uint256 requiredWeth = (nexaiAmount * wethWeight + 1e18 - 1) / 1e18;
          uint256 requiredUsdc = (nexaiAmount * usdcWeight + 1e18 - 1) / 1e18;
          require(maxUsdcIn > requiredUsdc, "maxUsdcIn too low to cover direct USDC leg");
  
          usdc.safeTransferFrom(msg.sender, address(this), maxUsdcIn);
  
          uint256 usdcBudgetForSwap = maxUsdcIn - requiredUsdc;
          usdc.forceApprove(address(swapRouter), usdcBudgetForSwap);
          uint256 usdcSpentOnSwap = swapRouter.exactOutputSingle(
              ISwapRouter02.ExactOutputSingleParams({
                  tokenIn: address(usdc),
                  tokenOut: address(weth),
                  fee: swapFee,
                  recipient: address(this),
                  amountOut: requiredWeth,
                  amountInMaximum: usdcBudgetForSwap,
                  sqrtPriceLimitX96: 0
              })
          );
          usdc.forceApprove(address(swapRouter), 0);
  
          weth.forceApprove(address(core), requiredWeth);
          usdc.forceApprove(address(core), requiredUsdc);
          core.deposit(nexaiAmount);
          weth.forceApprove(address(core), 0);
          usdc.forceApprove(address(core), 0);
  
          // core.deposit() minted NEXAI to address(this) (the Zapper is msg.sender from
          // Core's point of view) - forward it on to the actual depositor.
          IERC20(address(core.indexToken())).safeTransfer(msg.sender, nexaiAmount);
  
          usdcSpent = requiredUsdc + usdcSpentOnSwap;
          uint256 refund = maxUsdcIn - usdcSpent;
          if (refund > 0) usdc.safeTransfer(msg.sender, refund);
  
          emit ZappedDeposit(msg.sender, nexaiAmount, usdcSpent);

  import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
  import {Math} from "@openzeppelin/contracts/utils/math/Math.sol";
  import {MockERC20} from "../src/mocks/MockERC20.sol";
  
  /// @dev Minimal slice of Uniswap V3's NonfungiblePositionManager ABI - only what
  /// this script needs (pool creation/initialization + minting a full-range position).
  interface INonfungiblePositionManager {
      struct MintParams {
          address token0;
          address token1;
          uint24 fee;
          int24 tickLower;
          int24 tickUpper;
          uint256 amount0Desired;
          uint256 amount1Desired;
          uint256 amount0Min;
          uint256 amount1Min;
          address recipient;
          uint256 deadline;
      }
  
      function createAndInitializePoolIfNecessary(address token0, address token1, uint24 fee, uint160 sqrtPriceX96)
          external
          payable
          returns (address pool);
  
      function mint(MintParams calldata params)
          external
          payable
          returns (uint256 tokenId, uint128 liquidity, uint256 amount0, uint256 amount1);
  }
  
  /**
   * @title SeedPool
   * @dev Creates and seeds a Uniswap V3 pool for the mock WETH/USDC pair on Base Sepolia,
   * so NexaiZapper has real liquidity to swap against. Testnet-only.
   */
  contract SeedPoolScript is Script {
      address constant POSITION_MANAGER = 0x27F971cb582BF9E50F397e4d29a5C7A34f11faA2;
  
      address constant WETH = 0x5dAa95Cc547e98B3CEf227eD1D612CDd3d9E30a6;
      address constant USDC = 0x1a0e689b2b55c4E92571b37e0DA2C3b0EeE30B4a;
  
      uint24 constant FEE = 3000; // 0.3%, standard tier for a non-stablecoin pair
  
      // Full-range ticks for the 0.3% tier (tick spacing 60): floor(887272 / 60) * 60
      int24 constant TICK_LOWER = -887220;
      int24 constant TICK_UPPER = 887220;
  
      // Seed liquidity at the same 0.5 WETH : 1000 USDC ratio used by the basket weights,
      // scaled up 40x so the pool has enough depth for realistic zapper test swaps.
      uint256 constant WETH_SEED_AMOUNT = 20 * 10 ** 18;
      uint256 constant USDC_SEED_AMOUNT = 40_000 * 10 ** 6;
  
      function run() external {
          uint256 deployerPrivateKey = vm.envUint("PRIVATE_KEY");
          address deployer = vm.addr(deployerPrivateKey);
          vm.startBroadcast(deployerPrivateKey);
  
          // Mint the seed liquidity to the deployer (mock tokens, so this is free on testnet).
          MockERC20(WETH).mint(deployer, WETH_SEED_AMOUNT);
          MockERC20(USDC).mint(deployer, USDC_SEED_AMOUNT);
  
          // Uniswap V3 pools are keyed by the numerically-sorted token pair, and price is
          // always expressed as token1 per token0 - so we sort first and assign reserves
          // to whichever slot each token actually lands in, rather than assuming an order.
          (address token0, address token1) = WETH < USDC ? (WETH, USDC) : (USDC, WETH);
          (uint256 reserve0, uint256 reserve1) =
              token0 == WETH ? (WETH_SEED_AMOUNT, USDC_SEED_AMOUNT) : (USDC_SEED_AMOUNT, WETH_SEED_AMOUNT);
  
          // sqrtPriceX96 = sqrt(price) * 2^96, and price = reserve1/reserve0 in raw token
          // units. Computing it as sqrt(reserve1 * 2^192 / reserve0) via Math.mulDiv keeps
          // the intermediate product inside a 512-bit accumulator so it can't overflow
          // uint256, even though reserve1 * 2^192 alone would.
          uint160 sqrtPriceX96 = uint160(Math.sqrt(Math.mulDiv(reserve1, uint256(1) << 192, reserve0)));
  
          INonfungiblePositionManager positionManager = INonfungiblePositionManager(POSITION_MANAGER);
  
          address pool = positionManager.createAndInitializePoolIfNecessary(token0, token1, FEE, sqrtPriceX96);
          console.log("Pool created/initialized at:", pool);
  
          IERC20(token0).approve(POSITION_MANAGER, reserve0);
          IERC20(token1).approve(POSITION_MANAGER, reserve1);
  
          (uint256 tokenId, uint128 liquidity, uint256 amount0, uint256 amount1) = positionManager.mint(
              INonfungiblePositionManager.MintParams({
                  token0: token0,
                  token1: token1,
                  fee: FEE,
                  tickLower: TICK_LOWER,
                  tickUpper: TICK_UPPER,
                  amount0Desired: reserve0,
                  amount1Desired: reserve1,
                  amount0Min: 0,
                  amount1Min: 0,
                  recipient: deployer,
                  deadline: block.timestamp + 15 minutes
              })
          );
  
          console.log("Position NFT minted, tokenId:", tokenId);
          console.log("Liquidity:", uint256(liquidity));
          console.log("Amount0 used:", amount0);
          console.log("Amount1 used:", amount1);
  
          vm.stopBroadcast();
      }
  }

  # Frontend: One-Click USDC Deposit via Zapper
  
  The zapper is now deployed and verified working on Base Sepolia:
  `0xAE8645F73CB2dC48c5D531b5df62eC301bD54038`
  
  Update the Deposit tab so users can mint NEXAI with a single USDC
  approval instead of separately approving WETH and USDC.
  
  ## Requirements
  
  - Add `VITE_NEXAI_ZAPPER=0xAE8645F73CB2dC48c5D531b5df62eC301bD54038` to
    `.env` and wire it up alongside the other contract addresses
  - On the Deposit tab, offer this as the primary/default path (it's the
    actual "one-click" experience Dmitry asked for) — the existing
    two-token direct deposit can stay available as a secondary/advanced
    option, but shouldn't be the first thing users see
  - Flow: user enters desired NEXAI amount → single "Approve USDC" step
    → single "Deposit" step calling `zapDepositUsdc(nexaiAmount, maxUsdcIn)`
  - **`maxUsdcIn` needs a slippage buffer**, not the exact calculated
    cost — real Uniswap swaps have price movement between when the user
    sees a quote and when the transaction executes. Suggest computing the
    expected direct+swap cost (same math the zapper uses internally:
    read `basketWeights` for the round-up formula, direct USDC leg is
    fixed) and adding a buffer on top (e.g. 2-3%) for `maxUsdcIn`, so
    minor price movement doesn't cause an unnecessary revert
  - Live preview should still show the estimated USDC cost as the user
    types an amount, same pattern as the existing basket breakdown
    preview — just simplified to "approximately X USDC" instead of
    showing both WETH and USDC lines, since the user only needs to know
    what they're spending
  - Standard transaction status feedback (pending/success/error), same
    pattern as existing flows
  - After a successful zap deposit, refresh balances (NEXAI up, USDC
    down) — should work automatically if using the same live-updating
    balance hooks as the rest of the UI
  
  ## Out of Scope
  - Don't remove the existing direct WETH+USDC deposit flow — keep it as
    a secondary option, just not the default
  - No need to expose `maxUsdcIn` as a user-adjustable slippage setting
    for this trial — a sensible hardcoded buffer is fine

  Frontend's done, typechecked and built clean. Here's the walkthrough:
  
  Wiring — added VITE_NEXAI_ZAPPER to frontend/.env, and ZAPPER_ABI (just the one function it needs) plus the address to constants.ts.
  
  The cost estimate — the interesting part. There's no live Uniswap quote wired up here (that'd mean adding a Quoter contract call), so instead the estimate leans on data already on the page: the basket weights. Since the pool was seeded at the same ratio the basket weights encode (0.5 WETH : 1000 USDC), I used that ratio as a stand-in price — estimatedSwapCostUsdc = requiredWeth * (usdcWeight / wethWeight). Worth knowing: this simplifies to almost exactly requiredUsdc again, so the estimate reads as "total ≈ 2× the direct USDC leg." It's a reasonable approximation because we control the seed price, but it will drift from the real swap cost as soon as the pool gets traded and its price moves away from that starting ratio — at that point this stops being a good estimate and would need a real quoter. Flagging that now since it's a simplification, not a permanent design.
  
  The 3% buffer sits on top of that estimate for maxUsdcIn — this is what's actually approved and passed to the contract, not the number shown to the user (the UI shows the plain estimate, "~X USDC," so it doesn't look inflated). Any unspent buffer gets refunded by the Zapper itself.
  
  UI structure — depositMode state defaults to 'zap'; the old two-approval flow still exists verbatim, just gated behind an "Advanced" text link, with a matching link back. Both modes share the same NEXAI-amount input. Balance refresh needed no new code — the existing useReadContracts poll (every 5s) already covers it.
  
  Still open: the actual forge test run for the zapper contract, still blocked in this sandbox — that one's on you locally.

        { address: CONTRACT_ADDRESSES.WETH, abi: ERC20_ABI, functionName: 'allowance', args: [address || '0x0000000000000000000000000000000000000000', CONTRACT_ADDRESSES.NexaiCore] },
        { address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'allowance', args: [address || '0x0000000000000000000000000000000000000000', CONTRACT_ADDRESSES.NexaiCore] },
        { address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'allowance', args: [address || '0x0000000000000000000000000000000000000000', CONTRACT_ADDRESSES.NexaiZapper] },
      ],
      query: {
        enabled: isConnected,
        refetchInterval: 5000,
      }
    })
  
    const wethWeight = (reads?.[0]?.result as bigint) || 0n
    const usdcWeight = (reads?.[1]?.result as bigint) || 0n
    const wethBal = (reads?.[2]?.result as bigint) || 0n
    const usdcBal = (reads?.[3]?.result as bigint) || 0n
    const nexaiBal = (reads?.[4]?.result as bigint) || 0n
    const wethAllowance = (reads?.[5]?.result as bigint) || 0n
    const usdcAllowance = (reads?.[6]?.result as bigint) || 0n
    const usdcZapperAllowance = (reads?.[7]?.result as bigint) || 0n
  
    const { writeContract, data: txHash, isPending: isWriting } = useWriteContract()
    const { isLoading: isWaiting, isSuccess } = useWaitForTransactionReceipt({ hash: txHash })
  
    // Faucet — two separate write hooks so we can chain WETH → USDC
    const { writeContract: faucetWrite, data: faucetHash, isPending: isFaucetWriting } = useWriteContract()
    const [faucetStep, setFaucetStep] = useState<'idle' | 'weth' | 'usdc' | 'done' | 'error'>('idle')
    const { isLoading: isFaucetWaiting, isSuccess: isFaucetTxSuccess } = useWaitForTransactionReceipt({ hash: faucetHash })
  
    // Chain WETH → USDC mint using an effect so we never mutate state during render
    useEffect(() => {
      if (faucetStep === 'weth' && isFaucetTxSuccess && address) {
        setFaucetStep('usdc')
        faucetWrite({
          address: CONTRACT_ADDRESSES.USDC,
          abi: ERC20_ABI,
          functionName: 'mint',
          args: [address, 200000n * 10n ** 6n],
        })
      }
      if (faucetStep === 'usdc' && isFaucetTxSuccess) {
        setFaucetStep('done')
      }
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [isFaucetTxSuccess])
  
    const handleFaucet = useCallback(() => {
      if (!address) return
      setFaucetStep('weth')
      faucetWrite({
        address: CONTRACT_ADDRESSES.WETH,
        abi: ERC20_ABI,
        functionName: 'mint',
        args: [address, 100n * 10n ** 18n],
      })
    }, [address, faucetWrite])
  
    const isFaucetBusy = isFaucetWriting || isFaucetWaiting
    // Soft-disable once wallet already has ≥10 WETH (discourages spam but doesn't block)
    const alreadyFunded = wethBal >= 10n * 10n ** 18n
  
    const faucetStatus =
      faucetStep === 'weth' && (isFaucetWriting || isFaucetWaiting) ? '⏳ Minting WETH…' :
      faucetStep === 'usdc' && (isFaucetWriting || isFaucetWaiting) ? '⏳ Minting USDC…' :
      faucetStep === 'done' ? '✅ Tokens sent! Check balances above.' :
      faucetStep === 'error' ? '❌ Something went wrong. Try again.' : ''
    
    // Calculate Deposit Requirements
    const parsedDeposit = useMemo(() => {
      try {
        return parseUnits(depositAmount, 18)
      } catch {
        return 0n
      }
    }, [depositAmount])
  
    const requiredWeth = useMemo(() => {
      return parsedDeposit > 0n ? (parsedDeposit * wethWeight + 10n**18n - 1n) / (10n**18n) : 0n
    }, [parsedDeposit, wethWeight])
  
    const requiredUsdc = useMemo(() => {
      return parsedDeposit > 0n ? (parsedDeposit * usdcWeight + 10n**18n - 1n) / (10n**18n) : 0n
    }, [parsedDeposit, usdcWeight])
  
    // The zapper buys 100% of requiredWeth via swap. We don't have a live quote (no
    // Uniswap Quoter wired up), so we approximate its USDC cost using the basket's own
    // weight ratio as an implied price - self-consistent since the pool was seeded at
    // that same ratio. This cancels out to roughly requiredUsdc when weth/usdc weights
    // stay proportional, i.e. total cost is "about 2x the direct USDC leg".
    const estimatedSwapCostUsdc = useMemo(() => {
      return requiredWeth > 0n && wethWeight > 0n ? (requiredWeth * usdcWeight) / wethWeight : 0n
    }, [requiredWeth, usdcWeight, wethWeight])
  
    const estimatedTotalUsdc = useMemo(() => requiredUsdc + estimatedSwapCostUsdc, [requiredUsdc, estimatedSwapCostUsdc])
  
    // 3% buffer on top of the estimate so minor price movement between quote and
    // execution doesn't cause an unnecessary revert. This is what actually gets
    // approved/pulled - the zapper refunds whatever it doesn't spend.
    const maxUsdcIn = useMemo(() => (estimatedTotalUsdc * 103n) / 100n, [estimatedTotalUsdc])
  
    // Calculate Redeem Previews
    const parsedRedeem = useMemo(() => {
      try {
        return parseUnits(redeemAmount, 18)
      } catch {
        return 0n
      }
    }, [redeemAmount])
  
    const returnWeth = useMemo(() => {
      return parsedRedeem > 0n ? (parsedRedeem * wethWeight) / (10n**18n) : 0n
    }, [parsedRedeem, wethWeight])
  
    const returnUsdc = useMemo(() => {
      return parsedRedeem > 0n ? (parsedRedeem * usdcWeight) / (10n**18n) : 0n
    }, [parsedRedeem, usdcWeight])
  
    const needsWethApprove = requiredWeth > wethAllowance
    const needsUsdcApprove = requiredUsdc > usdcAllowance
    const needsUsdcApproveForZap = maxUsdcIn > usdcZapperAllowance
  
    const handleApproveWeth = () => {
      writeContract({ address: CONTRACT_ADDRESSES.WETH, abi: ERC20_ABI, functionName: 'approve', args: [CONTRACT_ADDRESSES.NexaiCore, requiredWeth] })
    }
    const handleApproveUsdc = () => {
      writeContract({ address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'approve', args: [CONTRACT_ADDRESSES.NexaiCore, requiredUsdc] })
    }
    const handleDeposit = () => {
      writeContract({ address: CONTRACT_ADDRESSES.NexaiCore, abi: NEXAI_CORE_ABI, functionName: 'deposit', args: [parsedDeposit] })
    }
    const handleApproveUsdcForZap = () => {
      writeContract({ address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'approve', args: [CONTRACT_ADDRESSES.NexaiZapper, maxUsdcIn] })
    }
    const handleZapDeposit = () => {
      writeContract({ address: CONTRACT_ADDRESSES.NexaiZapper, abi: ZAPPER_ABI, functionName: 'zapDepositUsdc', args: [parsedDeposit, maxUsdcIn] })
    }
    const handleRedeem = () => {
      writeContract({ address: CONTRACT_ADDRESSES.NexaiCore, abi: NEXAI_CORE_ABI, functionName: 'redeem', args: [parsedRedeem] })
    }
  
    const txStatus = isWriting ? 'Confirming in wallet...' : isWaiting ? 'Mining transaction...' : isSuccess ? 'Success!' : ''
  
    return (
      <div className="min-h-screen bg-slate-50 px-4 py-6 sm:p-8 font-sans">
        <div className="max-w-4xl mx-auto space-y-6">
          
          {/* Header */}
          <div className="flex flex-col sm:flex-row sm:justify-between sm:items-center gap-3 bg-white p-4 rounded-xl shadow-sm border border-slate-100">
            <h1 className="text-2xl font-bold text-slate-800 tracking-tight">NEXAI <span className="text-slate-400 font-normal">Vault</span></h1>
            {isConnected ? (
              <div className="flex items-center justify-between sm:justify-end gap-3">
                <span className="text-xs sm:text-sm text-slate-500 font-mono bg-slate-100 px-3 py-1 rounded-full truncate max-w-[160px] sm:max-w-none">
                  {address?.slice(0, 6)}...{address?.slice(-4)}
                </span>
                <Button variant="outline" size="sm" onClick={() => disconnect()}>Disconnect</Button>
              </div>
            ) : (
              <Button className="w-full sm:w-auto" onClick={() => connect({ connector: connectors[0] })}>Connect Wallet</Button>
            )}
          </div>
  
          {isConnected && (
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4 sm:gap-6">
              
              {/* Balances Panel */}
              <div className="col-span-1 space-y-4">
                <Card>
                  <CardHeader>
                    <CardTitle>Your Balances</CardTitle>
                    <CardDescription>Live wallet holdings</CardDescription>
                  </CardHeader>
                  <CardContent className="space-y-4">
                    <div className="flex justify-between items-center border-b pb-2">
                      <span className="text-slate-500 font-medium">NEXAI</span>
                      <span className="text-lg sm:text-xl font-bold truncate ml-2 max-w-[120px] sm:max-w-none text-right">{formatUnits(nexaiBal, 18)}</span>
                    </div>
                    <div className="flex justify-between items-center border-b pb-2">
                      <span className="text-slate-500 font-medium">WETH</span>
                      <span className="text-base sm:text-lg font-semibold truncate ml-2 max-w-[120px] sm:max-w-none text-right">{formatUnits(wethBal, 18)}</span>
                    </div>
                    <div className="flex justify-between items-center">
                      <span className="text-slate-500 font-medium">USDC</span>
                      <span className="text-base sm:text-lg font-semibold truncate ml-2 max-w-[120px] sm:max-w-none text-right">{formatUnits(usdcBal, 6)}</span>
                    </div>
                  </CardContent>
                </Card>
  
                {/* Faucet Card */}
                <Card className="border-dashed border-slate-300">
                  <CardHeader className="pb-3">
                    <CardTitle className="text-base">🚰 Testnet Faucet</CardTitle>
                    <CardDescription className="text-xs">
                      Get 100 WETH + 200,000 USDC sent to your wallet instantly.
                    </CardDescription>
                  </CardHeader>
                  <CardContent className="space-y-3">
                    {faucetStatus && (
                      <div className={`text-xs font-medium p-2 rounded ${
                        faucetStep === 'done' ? 'text-green-700 bg-green-50' :
                        faucetStep === 'error' ? 'text-red-700 bg-red-50' :
                        'text-blue-700 bg-blue-50'
                      }`}>
                        {faucetStatus}
                      </div>
                    )}
                    <Button
                      id="faucet-btn"
                      variant="outline"
                      className="w-full h-11 text-sm border-slate-300 hover:border-slate-400"
                      onClick={handleFaucet}
                      disabled={isFaucetBusy || alreadyFunded}
                      title={alreadyFunded ? 'You already have test tokens' : 'Mint test tokens to your wallet'}
                    >
                      {isFaucetBusy ? '⏳ Minting…' : alreadyFunded ? '✅ Already funded' : '🚰 Get Test Tokens'}
                    </Button>
                    {alreadyFunded && (
                      <p className="text-xs text-slate-400 text-center">Your wallet already holds ≥10 WETH.</p>
                    )}
                  </CardContent>
                </Card>
              </div>
  
              {/* Mint/Redeem Panel */}
              <div className="col-span-2">
                <Tabs defaultValue="deposit" className="w-full">
                  <TabsList className="grid w-full grid-cols-2">
                    <TabsTrigger value="deposit">Deposit & Mint</TabsTrigger>
                    <TabsTrigger value="redeem">Redeem & Burn</TabsTrigger>
                  </TabsList>
                  
                  <TabsContent value="deposit">
                    <Card>
                      <CardHeader>
                        <CardTitle>Mint NEXAI</CardTitle>
                        <CardDescription>
                          {depositMode === 'zap'
                            ? 'Deposit USDC only — the rest is swapped automatically to mint your NEXAI.'
                            : 'Deposit exact underlying assets to mint index tokens.'}
                        </CardDescription>
                      </CardHeader>
                      <CardContent className="space-y-5">
                        <div className="space-y-2">
                          <Label htmlFor="mint-amount">Amount of NEXAI to Mint</Label>
                          <Input
                            id="mint-amount"
                            placeholder="0.0"
                            inputMode="decimal"
                            value={depositAmount}
                            onChange={(e) => setDepositAmount(e.target.value)}
                            className="h-12 text-base"
                          />
                        </div>
  
                        {depositMode === 'zap' ? (
                          <>
                            {/* Live Cost Estimate */}
                            <div className="bg-slate-50 p-3 sm:p-4 rounded-lg border space-y-3">
                              <Label className="text-xs text-slate-500 uppercase tracking-wider">Estimated Cost</Label>
                              <div className="flex justify-between items-center">
                                <span className="font-medium text-slate-700">USDC</span>
                                <span className="font-mono text-sm sm:text-base truncate ml-2 max-w-[160px] text-right">~{formatUnits(estimatedTotalUsdc, 6)}</span>
                              </div>
                            </div>
  
                            {txStatus && <div className="text-sm font-medium text-blue-600 bg-blue-50 p-3 rounded">{txStatus}</div>}
  
                            <div className="flex gap-2">
                              {needsUsdcApproveForZap ? (
                                <Button className="w-full h-12 text-base" onClick={handleApproveUsdcForZap} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                  Approve USDC
                                </Button>
                              ) : (
                                <Button className="w-full h-12 text-base" onClick={handleZapDeposit} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                  Deposit & Mint
                                </Button>
                              )}
                            </div>
  
                            <button
                              type="button"
                              className="text-xs text-slate-400 hover:text-slate-600 underline w-full text-center"
                              onClick={() => setDepositMode('advanced')}
                            >
                              Advanced: deposit exact WETH + USDC directly
                            </button>
                          </>
                        ) : (
                          <>
                            {/* Live Basket Breakdown */}
                            <div className="bg-slate-50 p-3 sm:p-4 rounded-lg border space-y-3">
                              <Label className="text-xs text-slate-500 uppercase tracking-wider">Required Basket Deposit</Label>
                              <div className="flex justify-between items-center">
                                <span className="font-medium text-slate-700">WETH</span>
                                <span className="font-mono text-sm sm:text-base truncate ml-2 max-w-[160px] text-right">{formatUnits(requiredWeth, 18)}</span>
                              </div>
                              <div className="flex justify-between items-center">
                                <span className="font-medium text-slate-700">USDC</span>
                                <span className="font-mono text-sm sm:text-base truncate ml-2 max-w-[160px] text-right">{formatUnits(requiredUsdc, 6)}</span>
                              </div>
                            </div>
  
                            {txStatus && <div className="text-sm font-medium text-blue-600 bg-blue-50 p-3 rounded">{txStatus}</div>}
  
                            <div className="flex gap-2">
                              {needsWethApprove ? (
                                <Button className="w-full h-12 text-base" onClick={handleApproveWeth} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                  Approve WETH
                                </Button>
                              ) : needsUsdcApprove ? (
                                <Button className="w-full h-12 text-base" onClick={handleApproveUsdc} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                  Approve USDC
                                </Button>
                              ) : (
                                <Button className="w-full h-12 text-base" onClick={handleDeposit} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                  Deposit & Mint
                                </Button>
                              )}
                            </div>
  
                            <button
                              type="button"
                              className="text-xs text-slate-400 hover:text-slate-600 underline w-full text-center"
                              onClick={() => setDepositMode('zap')}
                            >
                              Back to one-click USDC deposit
                            </button>
                          </>
                        )}
                      </CardContent>
                    </Card>
                  </TabsContent>
  
                  <TabsContent value="redeem">
                    <Card>
                      <CardHeader>
                        <CardTitle>Redeem NEXAI</CardTitle>
                        <CardDescription>Burn your NEXAI to receive the underlying basket.</CardDescription>
                      </CardHeader>
                      <CardContent className="space-y-5">
                        <div className="space-y-2">
                          <Label htmlFor="redeem-amount">Amount of NEXAI to Redeem</Label>
                          <Input 
                            id="redeem-amount" 
                            placeholder="0.0"
                            inputMode="decimal"
                            value={redeemAmount}
                            onChange={(e) => setRedeemAmount(e.target.value)}
                            className="h-12 text-base"
                          />
                        </div>
  
                        {/* Live Return Breakdown */}
                        <div className="bg-slate-50 p-3 sm:p-4 rounded-lg border space-y-3">
                          <Label className="text-xs text-slate-500 uppercase tracking-wider">You Will Receive</Label>
                          <div className="flex justify-between items-center">
                            <span className="font-medium text-slate-700">WETH</span>
                            <span className="font-mono text-sm sm:text-base text-green-600 truncate ml-2 max-w-[160px] text-right">+{formatUnits(returnWeth, 18)}</span>
                          </div>
                          <div className="flex justify-between items-center">
                            <span className="font-medium text-slate-700">USDC</span>
                            <span className="font-mono text-sm sm:text-base text-green-600 truncate ml-2 max-w-[160px] text-right">+{formatUnits(returnUsdc, 6)}</span>
                          </div>
                        </div>
  
                        {txStatus && <div className="text-sm font-medium text-blue-600 bg-blue-50 p-3 rounded">{txStatus}</div>}
  
                        <Button className="w-full h-12 text-base" onClick={handleRedeem} disabled={isWriting || isWaiting || parsedRedeem === 0n}>
                          Redeem & Burn
                        </Button>
                      </CardContent>
                    </Card>
                  </TabsContent>
                </Tabs>
              </div>
  
            </div>
          )}
  
        </div>
      </div>
    )
  }
