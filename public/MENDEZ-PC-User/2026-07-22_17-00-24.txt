  // SPDX-License-Identifier: MIT
  pragma solidity ^0.8.20;
  
  import {Test} from "forge-std/Test.sol";
  import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
  import {NexaiGovToken} from "../src/governance/NexaiGovToken.sol";
  import {ISablierLockup, ISablierComptroller, Lockup, LockupLinear} from "../src/interfaces/ISablierLockup.sol";
  
  /**
   * @title NexaiVestingTest
   * @dev Fork tests for the Team/Ecosystem vesting streams created via
   * script/DeployVesting.s.sol, against Sablier's real, deployed v4.0 SablierLockup
   * singleton on Base Sepolia - same "genuine contract, not a mock" reasoning as
   * NexaiCoreAdmin.t.sol's Safe fork tests, since the whole point of verifying the
   * address/interface/cancel semantics up front was to test against real behavior, not an
   * assumed one. Requires BASE_SEPOLIA_RPC_URL in `.env` - see README "Governance Token".
   *
   * `sender` below stands in for AdminSafe (DeployVesting.s.sol's real cancel-authority
   * holder): cancel-authorization is a plain address check on Sablier's side, independent
   * of how that address produces signatures, so a plain address keeps these tests focused
   * on vesting/cancel mechanics - the Safe multisig execution flow itself is already
   * covered by NexaiCoreAdmin.t.sol and isn't re-tested here.
   */
  contract NexaiVestingTest is Test {
      address constant SABLIER_LOCKUP = 0xA65A1ecEf02C2B9BDc604951c6A6D02335D4532e;
  
      NexaiGovToken public govToken;
  
      address public sender = address(this);
      address public recipient = address(0xB0B);
  
      uint40 constant CLIFF = 180 days;
      uint40 constant DURATION = 730 days;
      uint128 constant AMOUNT = 4_000_000 * 10 ** 18;
  
      function setUp() public {
          vm.createSelectFork(vm.envString("BASE_SEPOLIA_RPC_URL"));
  
          govToken = new NexaiGovToken(); // mints full supply to address(this), which also
              // holds ADMIN_ROLE - mirrors DeployVesting.s.sol's deployer.
  
          govToken.setCapExempt(SABLIER_LOCKUP, true);
          govToken.approve(SABLIER_LOCKUP, type(uint256).max);
      }
  
      function _createStream(IERC20 token, address to, uint128 amount, uint40 cliff, uint40 duration, bool cancelable)
          internal
          returns (uint256 streamId)
      {
          streamId = ISablierLockup(SABLIER_LOCKUP).createWithDurationsLL(
              Lockup.CreateWithDurations({
                  sender: sender,
                  recipient: to,
                  depositAmount: amount,
                  token: token,
                  cancelable: cancelable,
                  transferable: false,
                  shape: "Test Vesting"
              }),
              LockupLinear.UnlockAmounts({start: 0, cliff: 0}),
              1,
              LockupLinear.Durations({cliff: cliff, total: duration})
          );
      }
  
      // ------------------------------------------------------------------
      // 1. Streams are created with the right sender/recipient/amount/cancelability
      // ------------------------------------------------------------------
      function test_StreamCreated_WithCorrectParams() public {
          uint256 streamId = _createStream(govToken, recipient, AMOUNT, CLIFF, DURATION, true);
  
          assertEq(ISablierLockup(SABLIER_LOCKUP).getSender(streamId), sender, "wrong sender");
          assertEq(ISablierLockup(SABLIER_LOCKUP).getRecipient(streamId), recipient, "wrong recipient");
          assertEq(ISablierLockup(SABLIER_LOCKUP).getDepositedAmount(streamId), AMOUNT, "wrong deposit");
          assertTrue(ISablierLockup(SABLIER_LOCKUP).isCancelable(streamId), "should be cancelable");
      }
  
      // ------------------------------------------------------------------
      // 2. Team-shaped stream (with cliff): nothing vests before the cliff
      // ------------------------------------------------------------------
      function test_TeamStream_NothingVestsBeforeCliff() public {
          uint256 streamId = _createStream(govToken, recipient, AMOUNT, CLIFF, DURATION, true);
  
          assertEq(ISablierLockup(SABLIER_LOCKUP).streamedAmountOf(streamId), 0, "should be 0 at creation");
  
          vm.warp(block.timestamp + CLIFF - 1);
          assertEq(ISablierLockup(SABLIER_LOCKUP).streamedAmountOf(streamId), 0, "should still be 0 right before cliff");
      }
  
      // ------------------------------------------------------------------
      // 3. Team-shaped stream: linear from cliff to end, 100% at end
      // ------------------------------------------------------------------
      function test_TeamStream_LinearAfterCliff_FullyVestedAtEnd() public {
          uint256 streamId = _createStream(govToken, recipient, AMOUNT, CLIFF, DURATION, true);
  
          uint256 streamableDuration = DURATION - CLIFF;
          vm.warp(block.timestamp + CLIFF + streamableDuration / 2);
          assertEq(
              ISablierLockup(SABLIER_LOCKUP).streamedAmountOf(streamId),
              AMOUNT / 2,
              "should be exactly half-vested halfway between cliff and end"
          );
  
          vm.warp(block.timestamp + streamableDuration); // now past the end timestamp
          assertEq(ISablierLockup(SABLIER_LOCKUP).streamedAmountOf(streamId), AMOUNT, "should be fully vested at end");
      }
  
      // ------------------------------------------------------------------
      // 4. Ecosystem-shaped stream (no cliff): vesting starts immediately
      // ------------------------------------------------------------------
      function test_EcosystemStream_NoCliff_VestsFromStart() public {
          uint256 streamId = _createStream(govToken, recipient, AMOUNT, 0, DURATION, true);
  
          vm.warp(block.timestamp + DURATION / 4);
          assertEq(
              ISablierLockup(SABLIER_LOCKUP).streamedAmountOf(streamId),
              AMOUNT / 4,
              "should already be a quarter-vested with no cliff gating it"
          );
      }
  
      // ------------------------------------------------------------------
      // 5. Cancellation: vested-but-unwithdrawn stays claimable by recipient, unvested
      //    is refunded to sender - the forfeiture-on-departure mechanism
      // ------------------------------------------------------------------
      function test_Cancel_RefundsUnvestedToSender_LeavesVestedClaimableByRecipient() public {
          uint256 streamId = _createStream(govToken, recipient, AMOUNT, CLIFF, DURATION, true);
  
          uint256 streamableDuration = DURATION - CLIFF;
          vm.warp(block.timestamp + CLIFF + streamableDuration / 4); // a quarter through vesting
  
          uint128 vestedAtCancel = ISablierLockup(SABLIER_LOCKUP).streamedAmountOf(streamId);
          assertEq(vestedAtCancel, AMOUNT / 4, "sanity check on vested amount before cancel");
  
          uint256 senderBalanceBefore = govToken.balanceOf(sender);
  
          uint128 refunded = ISablierLockup(SABLIER_LOCKUP).cancel(streamId);
  
          assertEq(refunded, AMOUNT - vestedAtCancel, "refund should be exactly the unvested remainder");
          assertEq(
              govToken.balanceOf(sender), senderBalanceBefore + refunded, "sender should receive the refund immediately"
          );
          assertTrue(ISablierLockup(SABLIER_LOCKUP).wasCanceled(streamId), "stream should be marked canceled");
          assertFalse(
              ISablierLockup(SABLIER_LOCKUP).isCancelable(streamId), "canceled stream shouldn't be cancelable again"
          );
  
          // The recipient can still claim what had already vested before the cancellation.
          // withdraw/withdrawMax charge a comptroller fee in ETH (verified against
          // SablierLockup.sol's source: create/cancel don't, withdraw does) - queried live
          // against the fork rather than assumed to be zero on testnet.
          ISablierComptroller comptroller = ISablierLockup(SABLIER_LOCKUP).comptroller();
          uint256 minFee = comptroller.calculateMinFeeWeiFor(ISablierComptroller.Protocol.Lockup, sender);
          vm.deal(recipient, minFee);
  
          vm.prank(recipient);
          uint128 withdrawn = ISablierLockup(SABLIER_LOCKUP).withdrawMax{value: minFee}(streamId, recipient);
  
          assertEq(withdrawn, vestedAtCancel, "recipient should be able to withdraw exactly what had vested");
          assertEq(govToken.balanceOf(recipient), vestedAtCancel, "recipient should receive their vested tokens");
      }
  
      // ------------------------------------------------------------------
      // 6. Only the stream's sender can cancel - not the recipient, not a stranger
      // ------------------------------------------------------------------
      function test_OnlySender_CanCancel() public {
          uint256 streamId = _createStream(govToken, recipient, AMOUNT, CLIFF, DURATION, true);
  
          vm.prank(recipient);
          vm.expectRevert();
          ISablierLockup(SABLIER_LOCKUP).cancel(streamId);
  
          address stranger = address(0xBAD);
          vm.prank(stranger);
          vm.expectRevert();
          ISablierLockup(SABLIER_LOCKUP).cancel(streamId);
      }
  
      // ------------------------------------------------------------------
      // 7. A non-cancelable stream can't be canceled even by its sender - proves
      //    cancelability is opt-in per stream, not implicit
      // ------------------------------------------------------------------
      function test_NonCancelableStream_CannotBeCanceled() public {
          uint256 streamId = _createStream(govToken, recipient, AMOUNT, CLIFF, DURATION, false);
  
          vm.expectRevert();
          ISablierLockup(SABLIER_LOCKUP).cancel(streamId);
      }
  
      // ------------------------------------------------------------------
      // 8. The 5% max-wallet cap would block funding Sablier without the exemption -
      //    proves the exemption in DeployVesting.s.sol is actually load-bearing
      // ------------------------------------------------------------------
      function test_CapExemption_IsRequiredBeforeFundingStreams() public {
          NexaiGovToken freshToken = new NexaiGovToken();
          freshToken.approve(SABLIER_LOCKUP, type(uint256).max);
          // Deliberately NOT exempting SABLIER_LOCKUP on freshToken yet.
  
          vm.expectRevert();
          _createStream(freshToken, recipient, AMOUNT, CLIFF, DURATION, true); // 4M > 5% of 100M supply
  
          freshToken.setCapExempt(SABLIER_LOCKUP, true);
  
          // Same call now succeeds once exempted.
          uint256 streamId = _createStream(freshToken, recipient, AMOUNT, CLIFF, DURATION, true);
          assertEq(ISablierLockup(SABLIER_LOCKUP).getDepositedAmount(streamId), AMOUNT);
      }
  }
