  script/TopUpPools.s.sol is ready. A few things worth calling out:
  
  One deliberate deviation from the ticket's literal wording: for the USDT/WETH top-up, rather than re-deriving its ratio from the WETH/USDC pool's price (as the original SeedUsdtPools.s.sol did when the pool didn't exist yet), this reads the USDT/WETH pool's own slot0() directly. Reasoning: that pool has now had its own independent trading history (it's the one drained to 40%), so its actual current price may have diverged from what the WETH/USDC pool implies. Since adding liquidity never moves a pool's price (only swaps do), using its own live price just means the top-up amounts land close to the actual current ratio instead of leaving a chunk refunded as mismatched dust. I generalized the existing _readUsdcPerWethFromLivePool helper into _readStablePerWethFromLivePool(stableToken, fee) so both pools reuse the identical sort/price-math logic — same pattern, parameterized rather than copy-pasted twice.
  
  Also added: a liquidity() read on each pool after minting, logged as "pool total in-range liquidity now" — satisfies "log the resulting approximate new total liquidity" concretely rather than just logging the amounts I intended to add.
  
  Confirmed your no-op claim: createAndInitializePoolIfNecessary only calls initialize() the first time a pool is created (Uniswap's periphery just returns the existing pool address if one's already there, since a pool can't be re-initialized) — so reusing that call here is safe, matching what you said.
  
  Same caveat as every contracts change so far: unverified by forge build/forge test here — worth running before broadcasting, given real funds (mock, but real transactions/gas) are on the line for three pool top-ups in one script.

  // SPDX-License-Identifier: MIT
  pragma solidity ^0.8.20;
  
  import {Script, console} from "forge-std/Script.sol";
  import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
  import {Math} from "@openzeppelin/contracts/utils/math/Math.sol";
  import {MockERC20} from "../src/mocks/MockERC20.sol";
  
  /// @dev Minimal slice of Uniswap V3's NonfungiblePositionManager ABI - same subset
  /// SeedPool.s.sol / SeedUsdtPools.s.sol use (pool creation/initialization + minting a
  /// full-range position).
  interface INonfungiblePositionManager {
      struct MintParams {
          address token0;
          address token1;
          uint24 fee;
          int24 tickLower;
          int24 tickUpper;
          uint256 amount0Desired;
          uint256 amount1Desired;
          uint256 amount0Min;
          uint256 amount1Min;
          address recipient;
          uint256 deadline;
      }
  
      function createAndInitializePoolIfNecessary(address token0, address token1, uint24 fee, uint160 sqrtPriceX96)
          external
          payable
          returns (address pool);
  
      function mint(MintParams calldata params)
          external
          payable
          returns (uint256 tokenId, uint128 liquidity, uint256 amount0, uint256 amount1);
  }
  
  interface IUniswapV3Factory {
      function getPool(address tokenA, address tokenB, uint24 fee) external view returns (address pool);
  }
  
  /// @dev Just enough of the pool's ABI to read its current price and total liquidity.
  interface IUniswapV3PoolState {
      function slot0()
          external
          view
          returns (
              uint160 sqrtPriceX96,
              int24 tick,
              uint16 observationIndex,
              uint16 observationCardinality,
              uint16 observationCardinalityNext,
              uint8 feeProtocol,
              bool unlocked
          );
  
      function liquidity() external view returns (uint128);
  }
  
  /**
   * @title TopUpPools
   * @dev Adds a fresh full-range liquidity position to each of the three existing zapper
   * pools (WETH/USDC, USDT/WETH, USDT/USDC), all priced off each pool's own *live*
   * slot0() rather than the original (now-drifted) seed ratios. Doesn't touch the
   * original seed positions or their NFTs - purely additive.
   * createAndInitializePoolIfNecessary is safe to call again here: Uniswap only calls
   * initialize() the first time a pool is created, so on an already-initialized pool it's
   * a no-op that just returns the existing pool address - the same helper below doubles
   * as both "create" and "top up" logic. Testnet only.
   */
  contract TopUpPoolsScript is Script {
      address constant FACTORY = 0x4752ba5DBc23f44D87826276BF6Fd6b1C372aD24;
      address constant POSITION_MANAGER = 0x27F971cb582BF9E50F397e4d29a5C7A34f11faA2;
  
      address constant WETH = 0x5dAa95Cc547e98B3CEf227eD1D612CDd3d9E30a6;
      address constant USDC = 0x1a0e689b2b55c4E92571b37e0DA2C3b0EeE30B4a;
      address constant USDT = 0x6972023ed96b910BBD6172c21dd0F86375419202;
  
      uint24 constant WETH_USDC_FEE = 3000;
      uint24 constant USDT_WETH_FEE = 3000;
      uint24 constant USDT_USDC_FEE = 500;
  
      // Full-range ticks, tick spacing 60 (0.3% tier): floor(887272 / 60) * 60
      int24 constant TICK_LOWER_3000 = -887220;
      int24 constant TICK_UPPER_3000 = 887220;
      // Full-range ticks, tick spacing 10 (0.05% tier): floor(887272 / 10) * 10
      int24 constant TICK_LOWER_500 = -887270;
      int24 constant TICK_UPPER_500 = 887270;
  
      // Top-up sizes: double each volatile pool's original seed depth. WETH/USDC was
      // drained to 63% and USDT/WETH to 40% of seed depth, so USDT/WETH gets a larger
      // top-up relative to its own original size. Stable pair gets a smaller, less
      // urgent bump since it's less prone to severe price impact.
      uint256 constant WETH_TOPUP_FOR_WETH_USDC_POOL = 40 * 10 ** 18; // 2x SeedPool.s.sol's original 20 WETH
      uint256 constant WETH_TOPUP_FOR_USDT_WETH_POOL = 20 * 10 ** 18; // 2x SeedUsdtPools.s.sol's original 10 WETH
      uint256 constant USDT_USDC_TOPUP = 50_000 * 10 ** 6; // equal amounts on both sides - 1:1
  
      function run() external {
          uint256 deployerPrivateKey = vm.envUint("PRIVATE_KEY");
          address deployer = vm.addr(deployerPrivateKey);
          vm.startBroadcast(deployerPrivateKey);
  
          // --- WETH/USDC top-up ---
          uint256 usdcPerWeth = _readStablePerWethFromLivePool(USDC, WETH_USDC_FEE);
          uint256 usdcTopup = (WETH_TOPUP_FOR_WETH_USDC_POOL * usdcPerWeth) / 10 ** 18;
          console.log("Live WETH/USDC price implies USDC per WETH:", usdcPerWeth);
  
          MockERC20(WETH).mint(deployer, WETH_TOPUP_FOR_WETH_USDC_POOL);
          MockERC20(USDC).mint(deployer, usdcTopup);
          console.log("Topping up WETH/USDC pool - WETH:", WETH_TOPUP_FOR_WETH_USDC_POOL);
          console.log("Topping up WETH/USDC pool - USDC:", usdcTopup);
          _addLiquidity(
              WETH,
              USDC,
              WETH_USDC_FEE,
              WETH_TOPUP_FOR_WETH_USDC_POOL,
              usdcTopup,
              TICK_LOWER_3000,
              TICK_UPPER_3000,
              deployer
          );
          _logPoolLiquidity("WETH/USDC", WETH, USDC, WETH_USDC_FEE);
  
          // --- USDT/WETH top-up. Priced off THIS pool's own live price, not derived from
          // the WETH/USDC pool - it has its own independent trading history and may have
          // drifted differently. ---
          uint256 usdtPerWeth = _readStablePerWethFromLivePool(USDT, USDT_WETH_FEE);
          uint256 usdtTopupForWethPool = (WETH_TOPUP_FOR_USDT_WETH_POOL * usdtPerWeth) / 10 ** 18;
          console.log("Live USDT/WETH price implies USDT per WETH:", usdtPerWeth);
  
          MockERC20(WETH).mint(deployer, WETH_TOPUP_FOR_USDT_WETH_POOL);
          MockERC20(USDT).mint(deployer, usdtTopupForWethPool);
          console.log("Topping up USDT/WETH pool - WETH:", WETH_TOPUP_FOR_USDT_WETH_POOL);
          console.log("Topping up USDT/WETH pool - USDT:", usdtTopupForWethPool);
          _addLiquidity(
              USDT,
              WETH,
              USDT_WETH_FEE,
              usdtTopupForWethPool,
              WETH_TOPUP_FOR_USDT_WETH_POOL,
              TICK_LOWER_3000,
              TICK_UPPER_3000,
              deployer
          );
          _logPoolLiquidity("USDT/WETH", USDT, WETH, USDT_WETH_FEE);
  
          // --- USDT/USDC top-up. Stays 1:1. ---
          MockERC20(USDT).mint(deployer, USDT_USDC_TOPUP);
          MockERC20(USDC).mint(deployer, USDT_USDC_TOPUP);
          console.log("Topping up USDT/USDC pool - USDT:", USDT_USDC_TOPUP);
          console.log("Topping up USDT/USDC pool - USDC:", USDT_USDC_TOPUP);
          _addLiquidity(
              USDT, USDC, USDT_USDC_FEE, USDT_USDC_TOPUP, USDT_USDC_TOPUP, TICK_LOWER_500, TICK_UPPER_500, deployer
          );
          _logPoolLiquidity("USDT/USDC", USDT, USDC, USDT_USDC_FEE);
  
          vm.stopBroadcast();
      }
  
      /// @dev Reads a WETH/stablecoin pool's current price and returns how much of the
      /// stablecoin (6dp) is implied per 1 whole WETH right now. Shared by both the
      /// WETH/USDC and USDT/WETH top-ups - each call reads that specific pool's own live
      /// price rather than assuming one pool's price applies to the other.
      function _readStablePerWethFromLivePool(address stableToken, uint24 fee) internal view returns (uint256) {
          address pool = IUniswapV3Factory(FACTORY).getPool(WETH, stableToken, fee);
          require(pool != address(0), "Pool not found - seed it first");
  
          (uint160 sqrtPriceX96,,,,,,) = IUniswapV3PoolState(pool).slot0();
          uint256 priceX192 = uint256(sqrtPriceX96) * uint256(sqrtPriceX96); // (token1/token0) * 2^192
  
          (address token0,) = WETH < stableToken ? (WETH, stableToken) : (stableToken, WETH);
  
          if (token0 == stableToken) {
              // priceX192 = WETH_wei per stable_wei * 2^192 -> invert for stable_wei per 1e18 WETH_wei
              return Math.mulDiv(10 ** 18, uint256(1) << 192, priceX192);
          } else {
              // priceX192 = stable_wei per WETH_wei * 2^192 already
              return Math.mulDiv(priceX192, 10 ** 18, uint256(1) << 192);
          }
      }
  
      /// @dev Same sort-then-assign-reserves-then-mint pattern as SeedPool.s.sol /
      /// SeedUsdtPools.s.sol, generalized to any pair. Adds a brand new position on top
      /// of whatever liquidity already exists - doesn't touch the original position.
      function _addLiquidity(
          address tokenA,
          address tokenB,
          uint24 fee,
          uint256 amountA,
          uint256 amountB,
          int24 tickLower,
          int24 tickUpper,
          address recipient
      ) internal {
          (address token0, address token1) = tokenA < tokenB ? (tokenA, tokenB) : (tokenB, tokenA);
          (uint256 reserve0, uint256 reserve1) = token0 == tokenA ? (amountA, amountB) : (amountB, amountA);
  
          uint160 sqrtPriceX96 = uint160(Math.sqrt(Math.mulDiv(reserve1, uint256(1) << 192, reserve0)));
  
          INonfungiblePositionManager positionManager = INonfungiblePositionManager(POSITION_MANAGER);
          positionManager.createAndInitializePoolIfNecessary(token0, token1, fee, sqrtPriceX96);
  
          IERC20(token0).approve(POSITION_MANAGER, reserve0);
          IERC20(token1).approve(POSITION_MANAGER, reserve1);
  
          (uint256 tokenId, uint128 liquidity, uint256 amount0, uint256 amount1) = positionManager.mint(
              INonfungiblePositionManager.MintParams({
                  token0: token0,
                  token1: token1,
                  fee: fee,
                  tickLower: tickLower,
                  tickUpper: tickUpper,
                  amount0Desired: reserve0,
                  amount1Desired: reserve1,
                  amount0Min: 0,
                  amount1Min: 0,
                  recipient: recipient,
                  deadline: block.timestamp + 15 minutes
              })
          );
  
          console.log("New position NFT minted, tokenId:", tokenId);
          console.log("Liquidity added:", uint256(liquidity));
          console.log("Amount0 used:", amount0);
          console.log("Amount1 used:", amount1);
      }
  
      function _logPoolLiquidity(string memory label, address tokenA, address tokenB, uint24 fee) internal view {
          address pool = IUniswapV3Factory(FACTORY).getPool(tokenA, tokenB, fee);
          uint128 totalLiquidity = IUniswapV3PoolState(pool).liquidity();
          console.log(string.concat(label, " pool total in-range liquidity now:"), uint256(totalLiquidity));
      }
  }
