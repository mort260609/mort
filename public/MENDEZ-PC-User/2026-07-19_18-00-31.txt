  -- Nexora System — Phase 1 schema
  -- Run this in the Supabase SQL editor for your project.
  -- Covers: users, user_stats, quests + Row Level Security.
  -- No AI-generated quests yet — quests.description just stores the raw
  -- onboarding goal text for tier='main'.
  
  -- ── users ────────────────────────────────────────────────────────────────
  -- One row per auth.users identity, keyed 1:1 by id.
  create table if not exists public.users (
    id uuid primary key references auth.users (id) on delete cascade,
    username text not null unique,
    name text,
    body_weight numeric,
    time_available_minutes integer,
    constraints text,
    level integer not null default 1,
    exp integer not null default 0,
    rank text not null default 'E',
    created_at timestamptz not null default now()
  );
  
  alter table public.users enable row level security;
  
  create policy "users_select_own"
    on public.users for select
    using (auth.uid() = id);
  
  create policy "users_insert_own"
    on public.users for insert
    with check (auth.uid() = id);
  
  create policy "users_update_own"
    on public.users for update
    using (auth.uid() = id)
    with check (auth.uid() = id);
  
  -- ── user_stats ───────────────────────────────────────────────────────────
  -- Core stat rows (STR/DEX/INT/VIT, ...), one row per stat per user.
  create table if not exists public.user_stats (
    id uuid primary key default gen_random_uuid(),
    user_id uuid not null references public.users (id) on delete cascade,
    stat_key text not null,
    stat_value integer not null default 5,
    created_at timestamptz not null default now(),
    unique (user_id, stat_key)
  );
  
  alter table public.user_stats enable row level security;
  
  create policy "user_stats_select_own"
    on public.user_stats for select
    using (auth.uid() = user_id);
  
  create policy "user_stats_insert_own"
    on public.user_stats for insert
    with check (auth.uid() = user_id);
  
  create policy "user_stats_update_own"
    on public.user_stats for update
    using (auth.uid() = user_id)
    with check (auth.uid() = user_id);
  
  -- ── quests ───────────────────────────────────────────────────────────────
  -- Phase 1 only writes a single tier='main' row holding the raw onboarding
  -- goal text. Side/Daily quest generation is Phase 2.
  create table if not exists public.quests (
    id uuid primary key default gen_random_uuid(),
    user_id uuid not null references public.users (id) on delete cascade,
    tier text not null check (tier in ('main', 'side', 'daily')),
    title text,
    description text,
    status text not null default 'active' check (status in ('active', 'completed', 'failed')),
    created_at timestamptz not null default now()
  );
  
  alter table public.quests enable row level security;
  
  create policy "quests_select_own"
    on public.quests for select
    using (auth.uid() = user_id);
  
  create policy "quests_insert_own"
    on public.quests for insert
    with check (auth.uid() = user_id);
  
  create policy "quests_update_own"
    on public.quests for update
    using (auth.uid() = user_id)
    with check (auth.uid() = user_id);

  -- Nexora System — Phase 1 Database Schema
  -- Tables: users, user_stats, quests
  -- Run this in the Supabase SQL editor after project creation.
  
  -- ============================================================
  -- USERS
  -- Extends Supabase's built-in auth.users with Nexora-specific profile data.
  -- ============================================================
  create table public.users (
    id uuid primary key references auth.users(id) on delete cascade,
    username text unique not null,
    name text not null,
    level int not null default 1,
    exp int not null default 0,
    exp_to_next_level int not null default 100,
    rank text not null default 'E' check (rank in ('E', 'D', 'C', 'B', 'A', 'S')),
    body_weight numeric,
    time_available_minutes int,           -- daily time constraint from onboarding
    main_quest_id uuid,                   -- set after main quest is created (fk added below)
    streak_count int not null default 0,
    streak_freezes_available int not null default 0,
    points_balance int not null default 0, -- NXR balance
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now()
  );
  
  -- ============================================================
  -- USER STATS
  -- Core RPG stats (STR, DEX, INT, VIT, WIS, etc.) — normalized so new
  -- stat types can be added without a schema migration.
  -- ============================================================
  create table public.user_stats (
    id uuid primary key default gen_random_uuid(),
    user_id uuid not null references public.users(id) on delete cascade,
    stat_key text not null,               -- e.g. 'STR', 'DEX', 'INT', 'VIT'
    value int not null default 0,
    updated_at timestamptz not null default now(),
    unique (user_id, stat_key)
  );
  
  -- ============================================================
  -- QUESTS
  -- Covers Main, Side, and Daily quests via the `tier` column.
  -- Only one active Main/Side/Daily quest per user is expected at a time —
  -- enforce that in application logic (or add a partial unique index later).
  -- ============================================================
  create table public.quests (
    id uuid primary key default gen_random_uuid(),
    user_id uuid not null references public.users(id) on delete cascade,
    tier text not null check (tier in ('main', 'side', 'daily')),
    title text not null,
    description text not null,
    status text not null default 'active'
      check (status in ('active', 'completed', 'failed', 'grace_period')),
    progress numeric not null default 0,   -- 0-100, used for Side Quest 50% grace rule
    reward_points int not null default 0,
    reward_exp int not null default 0,
    verification_method text not null default 'honor'
      check (verification_method in ('honor', 'ai_review', 'integration')),
    deadline timestamptz,                  -- null for Main Quest (open-ended until done)
    created_at timestamptz not null default now(),
    completed_at timestamptz
  );
  
  alter table public.users
    add constraint fk_main_quest foreign key (main_quest_id) references public.quests(id);
  
  -- ============================================================
  -- INDEXES
  -- ============================================================
  create index idx_quests_user_tier_status on public.quests (user_id, tier, status);
  create index idx_user_stats_user on public.user_stats (user_id);
  
  -- ============================================================
  -- ROW LEVEL SECURITY
  -- Users can only read/write their own data.
  -- ============================================================
  alter table public.users enable row level security;
  alter table public.user_stats enable row level security;
  alter table public.quests enable row level security;
  
  create policy "Users can view own profile" on public.users
    for select using (auth.uid() = id);
  create policy "Users can update own profile" on public.users
    for update using (auth.uid() = id);
  
  create policy "Users can view own stats" on public.user_stats
    for select using (auth.uid() = user_id);
  create policy "Users can update own stats" on public.user_stats
    for update using (auth.uid() = user_id);
  create policy "Users can insert own stats" on public.user_stats
    for insert with check (auth.uid() = user_id);
  
  create policy "Users can view own quests" on public.quests
    for select using (auth.uid() = user_id);
  create policy "Users can update own quests" on public.quests
    for update using (auth.uid() = user_id);
  create policy "Users can insert own quests" on public.quests
    for insert with check (auth.uid() = user_id);
  
  -- Note: quest generation (AI-created quests) should run through a
  -- server-side service role (bypassing RLS), not the client directly,
  -- since the AI Admin — not the player — creates quest rows.
