  import { useState, useMemo, useCallback, useEffect } from 'react'
  import { useAccount, useConnect, useDisconnect, useReadContracts, useSimulateContract, useWriteContract, useWaitForTransactionReceipt } from 'wagmi'
  import { parseUnits, formatUnits } from 'viem'
  import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card'
  import { Button } from '@/components/ui/button'
  import { Input } from '@/components/ui/input'
  import { Label } from '@/components/ui/label'
  import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs'
  import { CONTRACT_ADDRESSES, ERC20_ABI, NEXAI_CORE_ABI, ZAPPER_ABI, QUOTER_ABI } from './constants'
  
  export default function App() {
    const { address, isConnected } = useAccount()
    const { connectors, connect } = useConnect()
    const { disconnect } = useDisconnect()
  
    const [depositAmount, setDepositAmount] = useState('')
    const [redeemAmount, setRedeemAmount] = useState('')
    const [depositMode, setDepositMode] = useState<'zap' | 'advanced'>('zap')
    const [zapEntryToken, setZapEntryToken] = useState<'usdc' | 'weth'>('usdc')
  
    // Batch read contracts
    const { data: reads } = useReadContracts({
      contracts: [
        // Weights
        { address: CONTRACT_ADDRESSES.NexaiCore, abi: NEXAI_CORE_ABI, functionName: 'basketWeights', args: [CONTRACT_ADDRESSES.WETH] },
        { address: CONTRACT_ADDRESSES.NexaiCore, abi: NEXAI_CORE_ABI, functionName: 'basketWeights', args: [CONTRACT_ADDRESSES.USDC] },
        // Balances
        { address: CONTRACT_ADDRESSES.WETH, abi: ERC20_ABI, functionName: 'balanceOf', args: [address || '0x0000000000000000000000000000000000000000'] },
        { address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'balanceOf', args: [address || '0x0000000000000000000000000000000000000000'] },
        { address: CONTRACT_ADDRESSES.NexaiIndexToken, abi: ERC20_ABI, functionName: 'balanceOf', args: [address || '0x0000000000000000000000000000000000000000'] },
        // Allowances
        { address: CONTRACT_ADDRESSES.WETH, abi: ERC20_ABI, functionName: 'allowance', args: [address || '0x0000000000000000000000000000000000000000', CONTRACT_ADDRESSES.NexaiCore] },
        { address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'allowance', args: [address || '0x0000000000000000000000000000000000000000', CONTRACT_ADDRESSES.NexaiCore] },
        { address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'allowance', args: [address || '0x0000000000000000000000000000000000000000', CONTRACT_ADDRESSES.NexaiZapper] },
        { address: CONTRACT_ADDRESSES.WETH, abi: ERC20_ABI, functionName: 'allowance', args: [address || '0x0000000000000000000000000000000000000000', CONTRACT_ADDRESSES.NexaiZapper] },
        // Zapper's configured swap fee tier, read on-chain rather than duplicating the 3000 magic number here
        { address: CONTRACT_ADDRESSES.NexaiZapper, abi: ZAPPER_ABI, functionName: 'swapFee', args: [] },
      ],
      query: {
        enabled: isConnected,
        refetchInterval: 5000,
      }
    })
  
    const wethWeight = (reads?.[0]?.result as bigint) || 0n
    const usdcWeight = (reads?.[1]?.result as bigint) || 0n
    const wethBal = (reads?.[2]?.result as bigint) || 0n
    const usdcBal = (reads?.[3]?.result as bigint) || 0n
    const nexaiBal = (reads?.[4]?.result as bigint) || 0n
    const wethAllowance = (reads?.[5]?.result as bigint) || 0n
    const usdcAllowance = (reads?.[6]?.result as bigint) || 0n
    const usdcZapperAllowance = (reads?.[7]?.result as bigint) || 0n
    const wethZapperAllowance = (reads?.[8]?.result as bigint) || 0n
    const swapFee = (reads?.[9]?.result as number) || 3000
  
    const { writeContract, data: txHash, isPending: isWriting } = useWriteContract()
    const { isLoading: isWaiting, isSuccess } = useWaitForTransactionReceipt({ hash: txHash })
  
    // Faucet — two separate write hooks so we can chain WETH → USDC
    const { writeContract: faucetWrite, data: faucetHash, isPending: isFaucetWriting } = useWriteContract()
    const [faucetStep, setFaucetStep] = useState<'idle' | 'weth' | 'usdc' | 'done' | 'error'>('idle')
    const { isLoading: isFaucetWaiting, isSuccess: isFaucetTxSuccess } = useWaitForTransactionReceipt({ hash: faucetHash })
  
    // Chain WETH → USDC mint using an effect so we never mutate state during render
    useEffect(() => {
      if (faucetStep === 'weth' && isFaucetTxSuccess && address) {
        setFaucetStep('usdc')
        faucetWrite({
          address: CONTRACT_ADDRESSES.USDC,
          abi: ERC20_ABI,
          functionName: 'mint',
          args: [address, 200000n * 10n ** 6n],
        })
      }
      if (faucetStep === 'usdc' && isFaucetTxSuccess) {
        setFaucetStep('done')
      }
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [isFaucetTxSuccess])
  
    const handleFaucet = useCallback(() => {
      if (!address) return
      setFaucetStep('weth')
      faucetWrite({
        address: CONTRACT_ADDRESSES.WETH,
        abi: ERC20_ABI,
        functionName: 'mint',
        args: [address, 100n * 10n ** 18n],
      })
    }, [address, faucetWrite])
  
    const isFaucetBusy = isFaucetWriting || isFaucetWaiting
    // Soft-disable once wallet already has ≥10 WETH (discourages spam but doesn't block)
    const alreadyFunded = wethBal >= 10n * 10n ** 18n
  
    const faucetStatus =
      faucetStep === 'weth' && (isFaucetWriting || isFaucetWaiting) ? '⏳ Minting WETH…' :
      faucetStep === 'usdc' && (isFaucetWriting || isFaucetWaiting) ? '⏳ Minting USDC…' :
      faucetStep === 'done' ? '✅ Tokens sent! Check balances above.' :
      faucetStep === 'error' ? '❌ Something went wrong. Try again.' : ''
    
    // Calculate Deposit Requirements
    const parsedDeposit = useMemo(() => {
      try {
        return parseUnits(depositAmount, 18)
      } catch {
        return 0n
      }
    }, [depositAmount])
  
    const requiredWeth = useMemo(() => {
      return parsedDeposit > 0n ? (parsedDeposit * wethWeight + 10n**18n - 1n) / (10n**18n) : 0n
    }, [parsedDeposit, wethWeight])
  
    const requiredUsdc = useMemo(() => {
      return parsedDeposit > 0n ? (parsedDeposit * usdcWeight + 10n**18n - 1n) / (10n**18n) : 0n
    }, [parsedDeposit, usdcWeight])
  
    // The USDC-entry zap buys 100% of requiredWeth via swap. quoteExactOutputSingle is
    // the read-only counterpart to the exact-output swap the zapper actually performs, so
    // this asks the live pool "what would it cost right now to receive requiredWeth" -
    // an accurate current-price estimate, replacing the earlier basket-weight-ratio
    // approximation (which went stale as soon as the pool's price moved from its seed ratio).
    // Only enabled while USDC entry is selected, so we're not firing both quotes at once.
    const { data: usdcQuoteSim } = useSimulateContract({
      address: CONTRACT_ADDRESSES.QuoterV2,
      abi: QUOTER_ABI,
      functionName: 'quoteExactOutputSingle',
      args: [{
        tokenIn: CONTRACT_ADDRESSES.USDC,
        tokenOut: CONTRACT_ADDRESSES.WETH,
        amount: requiredWeth,
        fee: swapFee,
        sqrtPriceLimitX96: 0n,
      }],
      query: {
        enabled: zapEntryToken === 'usdc' && requiredWeth > 0n,
        refetchInterval: 5000,
      },
    })
  
    const estimatedSwapCostUsdc = (usdcQuoteSim?.result?.[0] as bigint | undefined) ?? 0n
  
    const estimatedTotalUsdc = useMemo(() => requiredUsdc + estimatedSwapCostUsdc, [requiredUsdc, estimatedSwapCostUsdc])
  
    // 5% buffer on top of the estimate. The estimate itself is now a real live quote, so
    // this only needs to cover ordinary price movement in the gap between the last quote
    // refresh (polled every 5s) and the moment the tx actually lands (wallet confirmation
    // + block time can easily add tens of seconds) - not compensate for a stale model like
    // the old weight-ratio approximation did. 5% is a judgment call for testnet, where
    // swings can be sharper than a real liquid market; revisit if it proves too tight.
    const maxUsdcIn = useMemo(() => (estimatedTotalUsdc * 105n) / 100n, [estimatedTotalUsdc])
  
    // Mirror of the USDC quote above, reversed: the WETH-entry zap buys requiredUsdc by
    // swapping WETH in, so we ask "what would it cost right now (in WETH) to receive
    // requiredUsdc". Only enabled while WETH entry is selected.
    const { data: wethQuoteSim } = useSimulateContract({
      address: CONTRACT_ADDRESSES.QuoterV2,
      abi: QUOTER_ABI,
      functionName: 'quoteExactOutputSingle',
      args: [{
        tokenIn: CONTRACT_ADDRESSES.WETH,
        tokenOut: CONTRACT_ADDRESSES.USDC,
        amount: requiredUsdc,
        fee: swapFee,
        sqrtPriceLimitX96: 0n,
      }],
      query: {
        enabled: zapEntryToken === 'weth' && requiredUsdc > 0n,
        refetchInterval: 5000,
      },
    })
  
    const estimatedSwapCostWeth = (wethQuoteSim?.result?.[0] as bigint | undefined) ?? 0n
  
    const estimatedTotalWeth = useMemo(() => requiredWeth + estimatedSwapCostWeth, [requiredWeth, estimatedSwapCostWeth])
  
    // Same 5% buffer, same reasoning as maxUsdcIn above.
    const maxWethIn = useMemo(() => (estimatedTotalWeth * 105n) / 100n, [estimatedTotalWeth])
  
    // Calculate Redeem Previews
    const parsedRedeem = useMemo(() => {
      try {
        return parseUnits(redeemAmount, 18)
      } catch {
        return 0n
      }
    }, [redeemAmount])
  
    const returnWeth = useMemo(() => {
      return parsedRedeem > 0n ? (parsedRedeem * wethWeight) / (10n**18n) : 0n
    }, [parsedRedeem, wethWeight])
  
    const returnUsdc = useMemo(() => {
      return parsedRedeem > 0n ? (parsedRedeem * usdcWeight) / (10n**18n) : 0n
    }, [parsedRedeem, usdcWeight])
  
    const needsWethApprove = requiredWeth > wethAllowance
    const needsUsdcApprove = requiredUsdc > usdcAllowance
    const needsUsdcApproveForZap = maxUsdcIn > usdcZapperAllowance
    const needsWethApproveForZap = maxWethIn > wethZapperAllowance
  
    const handleApproveWeth = () => {
      writeContract({ address: CONTRACT_ADDRESSES.WETH, abi: ERC20_ABI, functionName: 'approve', args: [CONTRACT_ADDRESSES.NexaiCore, requiredWeth] })
    }
    const handleApproveUsdc = () => {
      writeContract({ address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'approve', args: [CONTRACT_ADDRESSES.NexaiCore, requiredUsdc] })
    }
    const handleDeposit = () => {
      writeContract({ address: CONTRACT_ADDRESSES.NexaiCore, abi: NEXAI_CORE_ABI, functionName: 'deposit', args: [parsedDeposit] })
    }
    const handleApproveUsdcForZap = () => {
      writeContract({ address: CONTRACT_ADDRESSES.USDC, abi: ERC20_ABI, functionName: 'approve', args: [CONTRACT_ADDRESSES.NexaiZapper, maxUsdcIn] })
    }
    const handleZapDeposit = () => {
      writeContract({ address: CONTRACT_ADDRESSES.NexaiZapper, abi: ZAPPER_ABI, functionName: 'zapDepositUsdc', args: [parsedDeposit, maxUsdcIn] })
    }
    const handleApproveWethForZap = () => {
      writeContract({ address: CONTRACT_ADDRESSES.WETH, abi: ERC20_ABI, functionName: 'approve', args: [CONTRACT_ADDRESSES.NexaiZapper, maxWethIn] })
    }
    const handleZapDepositWeth = () => {
      writeContract({ address: CONTRACT_ADDRESSES.NexaiZapper, abi: ZAPPER_ABI, functionName: 'zapDepositWeth', args: [parsedDeposit, maxWethIn] })
    }
    const handleRedeem = () => {
      writeContract({ address: CONTRACT_ADDRESSES.NexaiCore, abi: NEXAI_CORE_ABI, functionName: 'redeem', args: [parsedRedeem] })
    }
  
    const txStatus = isWriting ? 'Confirming in wallet...' : isWaiting ? 'Mining transaction...' : isSuccess ? 'Success!' : ''
  
    return (
      <div className="min-h-screen bg-slate-50 px-4 py-6 sm:p-8 font-sans">
        <div className="max-w-4xl mx-auto space-y-6">
          
          {/* Header */}
          <div className="flex flex-col sm:flex-row sm:justify-between sm:items-center gap-3 bg-white p-4 rounded-xl shadow-sm border border-slate-100">
            <h1 className="text-2xl font-bold text-slate-800 tracking-tight">NEXAI <span className="text-slate-400 font-normal">Vault</span></h1>
            {isConnected ? (
              <div className="flex items-center justify-between sm:justify-end gap-3">
                <span className="text-xs sm:text-sm text-slate-500 font-mono bg-slate-100 px-3 py-1 rounded-full truncate max-w-[160px] sm:max-w-none">
                  {address?.slice(0, 6)}...{address?.slice(-4)}
                </span>
                <Button variant="outline" size="sm" onClick={() => disconnect()}>Disconnect</Button>
              </div>
            ) : (
              <Button className="w-full sm:w-auto" onClick={() => connect({ connector: connectors[0] })}>Connect Wallet</Button>
            )}
          </div>
  
          {isConnected && (
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4 sm:gap-6">
              
              {/* Balances Panel */}
              <div className="col-span-1 space-y-4">
                <Card>
                  <CardHeader>
                    <CardTitle>Your Balances</CardTitle>
                    <CardDescription>Live wallet holdings</CardDescription>
                  </CardHeader>
                  <CardContent className="space-y-4">
                    <div className="flex justify-between items-center border-b pb-2">
                      <span className="text-slate-500 font-medium">NEXAI</span>
                      <span className="text-lg sm:text-xl font-bold truncate ml-2 max-w-[120px] sm:max-w-none text-right">{formatUnits(nexaiBal, 18)}</span>
                    </div>
                    <div className="flex justify-between items-center border-b pb-2">
                      <span className="text-slate-500 font-medium">WETH</span>
                      <span className="text-base sm:text-lg font-semibold truncate ml-2 max-w-[120px] sm:max-w-none text-right">{formatUnits(wethBal, 18)}</span>
                    </div>
                    <div className="flex justify-between items-center">
                      <span className="text-slate-500 font-medium">USDC</span>
                      <span className="text-base sm:text-lg font-semibold truncate ml-2 max-w-[120px] sm:max-w-none text-right">{formatUnits(usdcBal, 6)}</span>
                    </div>
                  </CardContent>
                </Card>
  
                {/* Faucet Card */}
                <Card className="border-dashed border-slate-300">
                  <CardHeader className="pb-3">
                    <CardTitle className="text-base">🚰 Testnet Faucet</CardTitle>
                    <CardDescription className="text-xs">
                      Get 100 WETH + 200,000 USDC sent to your wallet instantly.
                    </CardDescription>
                  </CardHeader>
                  <CardContent className="space-y-3">
                    {faucetStatus && (
                      <div className={`text-xs font-medium p-2 rounded ${
                        faucetStep === 'done' ? 'text-green-700 bg-green-50' :
                        faucetStep === 'error' ? 'text-red-700 bg-red-50' :
                        'text-blue-700 bg-blue-50'
                      }`}>
                        {faucetStatus}
                      </div>
                    )}
                    <Button
                      id="faucet-btn"
                      variant="outline"
                      className="w-full h-11 text-sm border-slate-300 hover:border-slate-400"
                      onClick={handleFaucet}
                      disabled={isFaucetBusy || alreadyFunded}
                      title={alreadyFunded ? 'You already have test tokens' : 'Mint test tokens to your wallet'}
                    >
                      {isFaucetBusy ? '⏳ Minting…' : alreadyFunded ? '✅ Already funded' : '🚰 Get Test Tokens'}
                    </Button>
                    {alreadyFunded && (
                      <p className="text-xs text-slate-400 text-center">Your wallet already holds ≥10 WETH.</p>
                    )}
                  </CardContent>
                </Card>
              </div>
  
              {/* Mint/Redeem Panel */}
              <div className="col-span-2">
                <Tabs defaultValue="deposit" className="w-full">
                  <TabsList className="grid w-full grid-cols-2">
                    <TabsTrigger value="deposit">Deposit & Mint</TabsTrigger>
                    <TabsTrigger value="redeem">Redeem & Burn</TabsTrigger>
                  </TabsList>
                  
                  <TabsContent value="deposit">
                    <Card>
                      <CardHeader>
                        <CardTitle>Mint NEXAI</CardTitle>
                        <CardDescription>
                          {depositMode === 'zap'
                            ? `Deposit ${zapEntryToken === 'usdc' ? 'USDC' : 'WETH'} only — the rest is swapped automatically to mint your NEXAI.`
                            : 'Deposit exact underlying assets to mint index tokens.'}
                        </CardDescription>
                      </CardHeader>
                      <CardContent className="space-y-5">
                        <div className="space-y-2">
                          <Label htmlFor="mint-amount">Amount of NEXAI to Mint</Label>
                          <Input
                            id="mint-amount"
                            placeholder="0.0"
                            inputMode="decimal"
                            value={depositAmount}
                            onChange={(e) => setDepositAmount(e.target.value)}
                            className="h-12 text-base"
                          />
                        </div>
  
                        {depositMode === 'zap' ? (
                          <>
                            {/* Entry Token Selector */}
                            <div className="flex gap-2">
                              <Button
                                type="button"
                                variant={zapEntryToken === 'usdc' ? 'default' : 'outline'}
                                className="flex-1 h-10 text-sm"
                                onClick={() => setZapEntryToken('usdc')}
                              >
                                Pay with USDC
                              </Button>
                              <Button
                                type="button"
                                variant={zapEntryToken === 'weth' ? 'default' : 'outline'}
                                className="flex-1 h-10 text-sm"
                                onClick={() => setZapEntryToken('weth')}
                              >
                                Pay with WETH
                              </Button>
                            </div>
  
                            {/* Live Cost Estimate */}
                            <div className="bg-slate-50 p-3 sm:p-4 rounded-lg border space-y-3">
                              <Label className="text-xs text-slate-500 uppercase tracking-wider">Estimated Cost</Label>
                              {zapEntryToken === 'usdc' ? (
                                <div className="flex justify-between items-center">
                                  <span className="font-medium text-slate-700">USDC</span>
                                  <span className="font-mono text-sm sm:text-base truncate ml-2 max-w-[160px] text-right">~{formatUnits(estimatedTotalUsdc, 6)}</span>
                                </div>
                              ) : (
                                <div className="flex justify-between items-center">
                                  <span className="font-medium text-slate-700">WETH</span>
                                  <span className="font-mono text-sm sm:text-base truncate ml-2 max-w-[160px] text-right">~{formatUnits(estimatedTotalWeth, 18)}</span>
                                </div>
                              )}
                            </div>
  
                            {txStatus && <div className="text-sm font-medium text-blue-600 bg-blue-50 p-3 rounded">{txStatus}</div>}
  
                            <div className="flex gap-2">
                              {zapEntryToken === 'usdc' ? (
                                needsUsdcApproveForZap ? (
                                  <Button className="w-full h-12 text-base" onClick={handleApproveUsdcForZap} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                    Approve USDC
                                  </Button>
                                ) : (
                                  <Button className="w-full h-12 text-base" onClick={handleZapDeposit} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                    Deposit & Mint
                                  </Button>
                                )
                              ) : (
                                needsWethApproveForZap ? (
                                  <Button className="w-full h-12 text-base" onClick={handleApproveWethForZap} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                    Approve WETH
                                  </Button>
                                ) : (
                                  <Button className="w-full h-12 text-base" onClick={handleZapDepositWeth} disabled={isWriting || isWaiting || parsedDeposit === 0n}>
                                    Deposit & Mint
                                  </Button>
                                )
                              )}
                            </div>
  
                            <button

  # USDT Entry Token: Mock Deploy, Two Pools, Zapper Function
  
  USDT isn't a basket asset (the basket is still just WETH/USDC) — it's
  purely an entry token. That means a USDT deposit needs **two swaps**,
  not one direct leg + one swap like the USDC/WETH zappers: USDT →
  requiredWeth, and USDT → requiredUsdc, both via Uniswap.
  
  No official/reliable testnet USDT exists on Base Sepolia (checked —
  only random unverified community tokens), so this uses a mock token,
  same pattern as WETH/USDC.
  
  ## Part 1: Deploy Mock USDT
  
  New Foundry script, `script/DeployUsdt.s.sol` (or extend existing
  deploy tooling) — deploys a `MockERC20` USDT (6 decimals, matching
  real USDT convention), mints some to the deployer for pool seeding.
  
  ## Part 2: Seed Two Pools
  
  New script `script/SeedUsdtPools.s.sol`, same pattern as
  `SeedPool.s.sol`:
  - **USDT/WETH pool** — seed at a realistic ratio (USDT ≈ $1, so price
    roughly matches whatever WETH's current implied USD price is in the
    existing WETH/USDC pool)
  - **USDT/USDC pool** — seed near 1:1 (both are ~$1 stablecoins)
  
  Reuse the same token0/token1 sorting and `sqrtPriceX96` math pattern
  already proven in `SeedPool.s.sol` — don't reinvent it.
  
  ## Part 3: Add `zapDepositUsdt` to NexaiZapper.sol
