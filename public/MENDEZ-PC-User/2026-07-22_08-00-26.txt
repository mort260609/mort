  // SPDX-License-Identifier: MIT
  pragma solidity ^0.8.20;
  
  import {Script, console} from "forge-std/Script.sol";
  import {NexaiCore} from "../src/NexaiCore.sol";
  
  /**
   * @title MigrateAdminToTimelock
   * @dev ONE-WAY migration of NexaiCore's admin control from the deployer EOA to the
   * multisig/timelock structure (DeploySafes.s.sol + DeployTimelock.s.sol). Kept as its
   * own explicit script, separate from the original Deploy.s.sol, because this is a
   * significant and deliberately irreversible-from-the-EOA's-side action, not a routine
   * deploy step.
   *
   * Order matters and is chosen to avoid ever self-locking the EOA out mid-migration:
   *   1. Grant the new roles to their intended holders first
   *   2. Verify each grant actually landed
   *   3. Only then revoke the EOA's roles, ADMIN_ROLE before DEFAULT_ADMIN_ROLE - the
   *      EOA needs DEFAULT_ADMIN_ROLE (the role-admin for ADMIN_ROLE/GUARDIAN_ROLE) to
   *      still be able to fix anything if an earlier step were wrong, so it's the last
   *      thing given up.
   *
   * Scoped strictly to NexaiCore. Note: NexaiIndexToken's own DEFAULT_ADMIN_ROLE
   * (controlling its CORE_ROLE grants) is untouched by this and remains with the
   * deployer EOA - a separate, out-of-scope gap worth flagging to the client.
   * Testnet only.
   */
  contract MigrateAdminToTimelockScript is Script {
      // Set these to the addresses printed by DeployTimelock.s.sol / DeploySafes.s.sol.
      //
      // IMPORTANT: CORE must point at a NexaiCore deployed from the current source (with
      // Pausable/GUARDIAN_ROLE) - the NexaiCore already live on Base Sepolia at deploy
      // time of this script predates that code and has no pause()/GUARDIAN_ROLE() to
      // grant, so this deliberately does NOT default to that address. Redeploy via
      // Deploy.s.sol first if needed, then fill in the fresh address here.
      address constant CORE = address(0);
      address constant TIMELOCK = address(0);
      address constant GUARDIAN_SAFE = address(0);
  
      function run() external {
          require(CORE != address(0), "Set CORE to a NexaiCore deployed from the current (Pausable) source first");
          require(TIMELOCK != address(0), "Set TIMELOCK to DeployTimelock.s.sol's TimelockController address first");
          require(GUARDIAN_SAFE != address(0), "Set GUARDIAN_SAFE to DeploySafes.s.sol's GuardianSafe address first");
  
          NexaiCore core = NexaiCore(CORE);
  
          uint256 deployerPrivateKey = vm.envUint("PRIVATE_KEY");
          address deployer = vm.addr(deployerPrivateKey);
          vm.startBroadcast(deployerPrivateKey);
  
          // --- Step 1: grant new roles ---
          core.grantRole(core.DEFAULT_ADMIN_ROLE(), TIMELOCK);
          core.grantRole(core.ADMIN_ROLE(), TIMELOCK);
          core.grantRole(core.GUARDIAN_ROLE(), GUARDIAN_SAFE);
  
          // --- Step 2: verify before touching the EOA's own access ---
          require(core.hasRole(core.DEFAULT_ADMIN_ROLE(), TIMELOCK), "Timelock did not receive DEFAULT_ADMIN_ROLE");
          require(core.hasRole(core.ADMIN_ROLE(), TIMELOCK), "Timelock did not receive ADMIN_ROLE");
          require(core.hasRole(core.GUARDIAN_ROLE(), GUARDIAN_SAFE), "GuardianSafe did not receive GUARDIAN_ROLE");
          console.log("Verified: TimelockController holds DEFAULT_ADMIN_ROLE + ADMIN_ROLE on NexaiCore");
          console.log("Verified: GuardianSafe holds GUARDIAN_ROLE on NexaiCore");
  
          // --- Step 3: revoke the EOA's roles, ADMIN_ROLE first ---
          core.revokeRole(core.ADMIN_ROLE(), deployer);
          console.log("Revoked ADMIN_ROLE from deployer EOA:", deployer);
  
          core.revokeRole(core.DEFAULT_ADMIN_ROLE(), deployer);
          console.log("Revoked DEFAULT_ADMIN_ROLE from deployer EOA:", deployer);
          console.log("Migration complete - NexaiCore admin control now belongs to the timelock/guardian structure.");
  
          vm.stopBroadcast();
      }
  }
